 44    def _build_system_block(self, system: str) -> list[dict[str, Any]]:
 45        """Build a system block with ephemeral cache_control for prompt caching."""
 46        return [
 47            {
 48                "type": "text",

 50                "cache_control": {"type": "ephemeral"},
 51            }
 52        ]

 54    def _build_tools(self, tools: list[dict[str, Any]]) -> list[dict[str, Any]]:
 55        """Convert OpenAI-style tool dicts to Anthropic format, adding cache control."""
 56        anthropic_tools: list[dict[str, Any]] = []
 57        for i, tool in enumerate(tools):
 58            t: dict[str, Any] = {
 59                "name": tool.get("name") or tool.get("function", {}).get("name", ""),
 60                "description": tool.get("description")
 61                or tool.get("function", {}).get("description", ""),
 62                "input_schema": tool.get("input_schema")
 63                or tool.get("parameters")
 64                or tool.get("function", {}).get("parameters", {}),
 65            }
 66            # Apply cache_control to the last tool when there are more than 3
 67            if len(tools) > 3 and i == len(tools) - 1:
 68                t["cache_control"] = {"type": "ephemeral"}
 69            anthropic_tools.append(t)
 70        return anthropic_tools

 72    def _map_messages(
 73        self, messages: list[dict[str, Any]]
 74    ) -> list[dict[str, Any]]:
 75        """Convert generic message dicts to Anthropic-compatible format."""
 76        result: list[dict[str, Any]] = []
 77        for msg in messages:
 78            role = msg.get("role", "user")
 79            content = msg.get("content", "")
 80            # Skip system messages — handled separately
 81            if role == "system":
 82                continue
 83            if isinstance(content, str):
 84                result.append({"role": role, "content": content})
 85            else:
 86                result.append({"role": role, "content": content})
 87        return result
