 19
 20    Each run gets its own directory: ``{base_path}/{tenant_id}/{run_id}/``

 23    the workspace root raises ``PermissionError``.

 34        """

 36
 37        Creates all intermediate directories as needed.
 38        """

 44    def resolve(self, run_id: str, tenant_id: str, relative: str | Path) -> Path:
 45        """
 46        Resolve ``relative`` within the workspace, raising PermissionError
 47        if the resolved path escapes the workspace directory.
 48        """
 49        workspace = self._workspace_root(run_id, tenant_id)
 50        resolved = (workspace / relative).resolve()
 51
 52        try:
 53            resolved.relative_to(workspace.resolve())
 54        except ValueError:
 55            raise PermissionError(
 56                f"Path traversal attempt detected: '{relative}' resolves to "
 57                f"'{resolved}' which is outside workspace '{workspace}'"
 58            )
 59
 60        return resolved

 97    async def get_size(self, run_id: str, tenant_id: str) -> int:
 98        """Return total bytes used by the workspace."""

101            return 0

105            if path.is_file():
106                try:

 34class DockerSandbox:
 35    """
 36    Execute Python code (or arbitrary commands) inside an isolated Docker container.
 37
 38    Security controls:
 39    - Memory limit (default 256 MiB)
 40    - CPU limit (1 core)
 41    - Optional network isolation (default: no network)
 42    - Read/write mount of workspace only
 43
 44    Falls back to RestrictedPythonExecutor if Docker is unavailable.
 45    """
 46
 47    def __init__(
 48        self,
 49        image: str = "python:3.11-slim",
 50        memory_limit: str = "256m",
 51        timeout: float = 30.0,
 52        network: bool = False,
 53    ) -> None:
 54        self._image = image
 55        self._memory_limit = memory_limit
 56        self._timeout = timeout
 57        self._network = network
 58
 59    # ------------------------------------------------------------------
 60    # Public API
 61    # ------------------------------------------------------------------
 62
 63    async def run_code(self, code: str, workspace_path: Path) -> SandboxResult:
 64        """
 65        Write ``code`` to a temp file in the workspace, then execute it
 66        inside a Docker container.
 67        """
 68        run_filename = f"run_{uuid.uuid4().hex[:8]}.py"
 69        run_path = workspace_path / run_filename
 70
 71        try:
 72            run_path.write_text(code, encoding="utf-8")
 73        except OSError as exc:
 74            raise SandboxError(f"Failed to write code to workspace: {exc}")
 75
 76        cmd = [
 77            "docker", "run",
 78            "--rm",
 79            f"--memory={self._memory_limit}",
 80            "--memory-swap=-1",        # disable swap
 81            "--cpus=1",
 82            f"--network={'none' if not self._network else 'bridge'}",
 83            "--volume", f"{workspace_path.resolve()}:/sandbox:rw",
 84            "--workdir=/sandbox",
 85            "--user=nobody",           # non-root execution
 86            "--security-opt=no-new-privileges",
 87            self._image,
 88            "python", f"/sandbox/{run_filename}",
 89        ]
 90
 91        result = await self.run_command(cmd=cmd, workspace_path=workspace_path)
 92
 93        # Cleanup temp file
 94        try:
 95            run_path.unlink(missing_ok=True)
 96        except OSError:
 97            pass
 98
 99        return result
100
101    async def run_command(
102        self,
103        cmd: list[str],
104        workspace_path: Path,
105        env: dict[str, str] | None = None,
106    ) -> SandboxResult:
107        """
108        Execute an arbitrary command inside a Docker container.
109
110        The first argument should be "docker" if this is intended as a
111        Docker run; otherwise the command is executed directly.
112        """
113        start_time = time.monotonic()
114
115        try:
116            proc = await asyncio.create_subprocess_exec(
117                *cmd,
118                stdout=asyncio.subprocess.PIPE,
119                stderr=asyncio.subprocess.PIPE,
120                env=env,
121            )
122        except FileNotFoundError as exc:
123            raise SandboxError(
124                f"Command not found: {cmd[0]}. "
125                "Is Docker installed and in PATH?"
126            ) from exc
127
128        timed_out = False
129        try:
130            stdout_bytes, stderr_bytes = await asyncio.wait_for(
131                proc.communicate(), timeout=self._timeout
132            )
133        except asyncio.TimeoutError:
134            timed_out = True
135            try:
136                proc.kill()
137            except ProcessLookupError:
138                pass
139            stdout_bytes = b""
140            stderr_bytes = b"Execution timed out."
141
142        elapsed_ms = (time.monotonic() - start_time) * 1000
143
144        exit_code = proc.returncode if proc.returncode is not None else -1
145
146        return SandboxResult(
147            stdout=stdout_bytes.decode("utf-8", errors="replace"),
148            stderr=stderr_bytes.decode("utf-8", errors="replace"),
149            exit_code=exit_code,
150            timed_out=timed_out,
151            execution_time_ms=elapsed_ms,
152        )
153
154    @staticmethod
155    async def is_available() -> bool:
156        """Check if Docker daemon is accessible."""
157        try:
158            proc = await asyncio.create_subprocess_exec(
159                "docker", "info",
160                stdout=asyncio.subprocess.DEVNULL,
161                stderr=asyncio.subprocess.DEVNULL,
162            )
163            await asyncio.wait_for(proc.communicate(), timeout=5.0)
164            return proc.returncode == 0
165        except (asyncio.TimeoutError, FileNotFoundError, OSError):
166            return False

class CheckpointManager: 
 65class CheckpointManager:
 66    """
 67    Saves and restores agent run state for crash recovery.
 68
 69    Checkpoint files are written atomically:
 70    1. Serialise to a ``.tmp`` file.
 71    2. Rename to the final checkpoint filename (atomic on POSIX filesystems).
 72
 73    The workspace layout:
 74        ``{workspace_base}/{tenant_id}/{run_id}/checkpoint.json``
 75    """
 76
 77    def __init__(self, workspace_base: str | Path) -> None:
 78        self._workspace_base = Path(workspace_base)
 79
 80    # ------------------------------------------------------------------
 81    # Public API
 82    # ------------------------------------------------------------------
 83
 84    async def save(
 85        self,
 86        ctx: "AgentContext",
 87        history: list[Any],
 88    ) -> Path:
 89        """
 90        Serialise ``ctx`` and ``history`` to a checkpoint file.
 91
 92        Uses atomic write (tmp → rename) to prevent partial checkpoint reads
 93        on crash.
 94
 95        ``history`` should be a list of ConversationMessage objects or dicts.
 96        """
 97        checkpoint_dir = self._checkpoint_dir(ctx.run_id, ctx.tenant_id)
 98        checkpoint_dir.mkdir(parents=True, exist_ok=True)
 99
100        # Serialise history entries
101        history_snapshot: list[dict[str, Any]] = []
102        for msg in history:
103            if hasattr(msg, "to_dict"):
104                history_snapshot.append(msg.to_dict())
105            elif isinstance(msg, dict):
106                history_snapshot.append(msg)
107            else:
108                history_snapshot.append(
109                    {
110                        "role": getattr(msg, "role", "unknown"),
111                        "content": str(getattr(msg, "content", "")),
112                        "tokens": getattr(msg, "tokens", 0),
113                        "timestamp": getattr(
114                            msg, "timestamp", datetime.now(timezone.utc)
115                        ).isoformat()
116                        if hasattr(getattr(msg, "timestamp", None), "isoformat")
117                        else str(getattr(msg, "timestamp", "")),
118                    }
119                )
120
121        data = CheckpointData(
122            run_id=ctx.run_id,
123            tenant_id=ctx.tenant_id,
124            agent_type=ctx.agent_type,
125            task=ctx.task,
126            step_count=ctx.step_count,
127            token_count=ctx.token_count,
128            started_at=ctx.started_at,
129            metadata=dict(ctx.metadata),
130            failed=ctx.failed,
131            failure_class=ctx.failure_class,
132            history_snapshot=history_snapshot,
133        )
134
135        checkpoint_path = checkpoint_dir / _CHECKPOINT_FILENAME
136        tmp_path = checkpoint_path.with_suffix(_CHECKPOINT_TMP_SUFFIX)
137
138        try:
139            serialized = json.dumps(data.to_dict(), indent=2, default=str)
140            tmp_path.write_text(serialized, encoding="utf-8")
141            # Atomic rename
142            tmp_path.rename(checkpoint_path)
143            logger.debug("Checkpoint saved: %s", checkpoint_path)
144        except Exception as exc:
145            # Clean up tmp file if rename failed
146            try:
147                tmp_path.unlink(missing_ok=True)
148            except OSError:
149                pass
150            raise RuntimeError(f"Checkpoint save failed: {exc}") from exc
151
152        return checkpoint_path
153
154    async def load(
155        self,
156        run_id: str,
157        tenant_id: str,
158    ) -> CheckpointData | None:
159        """
160        Load and deserialise the checkpoint for ``run_id``.
161
162        Returns None if the checkpoint file does not exist.
163        """
164        checkpoint_path = self._checkpoint_path(run_id, tenant_id)
165
166        if not checkpoint_path.exists():
167            return None
168
169        try:
170            raw = checkpoint_path.read_text(encoding="utf-8")
171            data = json.loads(raw)
172            checkpoint = CheckpointData.from_dict(data)
173            logger.debug(
174                "Checkpoint loaded: run_id=%s step=%d",
175                run_id,
176                checkpoint.step_count,
177            )
178            return checkpoint
179        except (json.JSONDecodeError, KeyError, TypeError, ValueError) as exc:
180            logger.error(
181                "Checkpoint for run %s is corrupt: %s", run_id, exc
182            )
183            return None
184
185    async def exists(self, run_id: str, tenant_id: str) -> bool:
186        """Return True if a valid checkpoint file exists for this run."""
187        return self._checkpoint_path(run_id, tenant_id).exists()
188
189    async def delete(self, run_id: str, tenant_id: str) -> None:
190        """Remove the checkpoint file for this run."""
191        checkpoint_path = self._checkpoint_path(run_id, tenant_id)
192        try:
193            checkpoint_path.unlink(missing_ok=True)
194            logger.debug("Checkpoint deleted: %s", checkpoint_path)
195        except OSError as exc:
196            logger.warning("Failed to delete checkpoint for %s: %s", run_id, exc)

248                "bool": bool,
249                "list": list,
250                "dict": dict,
251                "set": set,
252                "tuple": tuple,
253                "isinstance": isinstance,
254                "type": type,
255                "repr": repr,
256                "True": True,

258                "None": None,
259            },
260        }
261
262        loop = asyncio.get_running_loop()
263        exit_code = 0
