 96    async def execute(self, ctx: AgentContext, call: ToolCall) -> ToolResult:
 97        """Execute a tool call with full validation, safety, timeout, and logging.
 98
 99        Steps:
100        1. Lookup tool — raise ToolError(TOOL_NOT_FOUND) if missing.
101        2. JSON Schema validate args — raise ToolError(TOOL_SCHEMA_ERROR) if fails.
102        3. Safety check via pipeline.check_step — raise SafetyViolation if blocked.
103        4. Execute with asyncio.timeout(tool.timeout_seconds).
104        5. Audit log the call.
105        6. Increment tool_calls_total counter.
106        7. On exception: wrap in ToolError(TOOL_EXEC_ERROR), record failure.
107        """
108        # 1. Lookup
109        tool = self._tools.get(call.name)
110        if tool is None:
111            raise ToolError(
112                f"Tool '{call.name}' not found in registry.",
113                tool_name=call.name,
114                failure_class=FailureClass.TOOL_NOT_FOUND,
115                context={"run_id": ctx.run_id, "available": list(self._tools.keys())},
116            )
117
118        # 2. Schema validation
119        if hasattr(tool, "input_schema") and tool.input_schema:
120            try:
121                jsonschema.validate(instance=call.args, schema=tool.input_schema)
122            except jsonschema.ValidationError as exc:
123                raise ToolError(
124                    f"Tool '{call.name}' arg schema validation failed: {exc.message}",
125                    tool_name=call.name,
126                    failure_class=FailureClass.TOOL_SCHEMA_ERROR,
127                    context={
128                        "run_id": ctx.run_id,
129                        "args": call.args,
130                        "validation_error": exc.message,
131                    },
132                ) from exc
133
134        # 3. Safety check
135        if self._safety_pipeline is not None:
136            try:
137                step_payload = {"tool_name": call.name, "args": call.args}
138                guard_result = await _maybe_await(
139                    self._safety_pipeline.check_step(step_payload)
140                )
141                if guard_result is not None and hasattr(guard_result, "blocked") and guard_result.blocked:
142                    raise SafetyViolation(
143                        f"Tool call '{call.name}' blocked by safety pipeline: "
144                        f"{getattr(guard_result, 'reason', 'policy violation')}",
145                        guard_source="tool_registry",
146                        failure_class=FailureClass.SAFETY_STEP,
147                        context={"run_id": ctx.run_id, "tool_name": call.name},
148                    )
149            except SafetyViolation:
150                raise
151            except Exception as exc:
152                logger.warning(
153                    "Safety pipeline check_step raised unexpected error: %s", exc
154                )
155
156        # 4. Execute with timeout
157        timeout_seconds = getattr(tool, "timeout_seconds", 30.0)
158        result: ToolResult
159        try:
160            async with asyncio.timeout(timeout_seconds):
161                result = await tool.execute(ctx, call.args)
162        except asyncio.TimeoutError:
163            error_msg = (
164                f"Tool '{call.name}' timed out after {timeout_seconds:.1f}s"
165            )
166            logger.warning(error_msg)
167            result = ToolResult(data=None, error=error_msg)
168            # Audit log failure
169            await self._audit(ctx, call, result)
170            self._increment_metric(call.name, success=False)
171            raise ToolError(
172                error_msg,
173                tool_name=call.name,
174                failure_class=FailureClass.TOOL_TIMEOUT,
175                context={"run_id": ctx.run_id, "timeout_seconds": timeout_seconds},
176            )
177        except ToolError:
178            raise
179        except SafetyViolation:
180            raise
181        except Exception as exc:
182            error_msg = f"Tool '{call.name}' execution failed: {exc}"
183            logger.exception("Tool '%s' raised unhandled exception", call.name)
184            result = ToolResult(data=None, error=error_msg)
185            await self._audit(ctx, call, result)
186            self._increment_metric(call.name, success=False)
187            raise ToolError(
188                error_msg,
189                tool_name=call.name,
190                failure_class=FailureClass.TOOL_EXEC_ERROR,
191                context={"run_id": ctx.run_id, "original_error": str(exc)},
192            ) from exc
193
194        # 5. Output schema validation (if tool declares output_schema)
195        if hasattr(tool, "output_schema") and tool.output_schema and not result.is_error:
196            try:
197                jsonschema.validate(instance=result.data, schema=tool.output_schema)
198            except jsonschema.ValidationError as exc:
199                logger.warning(
200                    "Tool '%s' output failed schema validation: %s",
201                    call.name,
202                    exc.message,
203                )
204                result = ToolResult(
205                    data=result.data,
206                    error=f"Output schema error: {exc.message}",
207                    metadata=result.metadata,
208                )
209
210        # 6. Audit log
211        await self._audit(ctx, call, result)
212
213        # 7. Metrics
214        self._increment_metric(call.name, success=not result.is_error)
215
216        return result

246    def _increment_metric(self, tool_name: str, *, success: bool) -> None:
247        """Increment the tool_calls_total Prometheus counter."""
248        if self._metrics is None:
249            return
250        try:
251            labels = {"tool_name": tool_name, "success": str(success).lower()}
252            counter = getattr(self._metrics, _TOOL_CALLS_METRIC, None)
253            if counter is not None:
254                counter.labels(**labels).inc()
255        except Exception as exc:

134        """Establish connection to the MCP server and discover available tools.
135
136        Returns a list of MCPToolWrapper instances, one per tool.
137        Raises HarnessError(MCP_CONNECT_ERROR) on connection failure.
138        """
139        try:
140            from mcp import ClientSession, StdioServerParameters
141            from mcp.client.stdio import stdio_client
142
143            if self._config.transport == "stdio":
144                if not self._config.command:
145                    raise HarnessError(
146                        f"MCP server '{self._config.name}' configured as stdio "
147                        "but no command was provided.",
148                        failure_class=FailureClass.MCP_CONNECT_ERROR,
149                    )
150                env = {**os.environ, **self._config.env}
151                server_params = StdioServerParameters(
152                    command=self._config.command[0],
153                    args=self._config.command[1:],
154                    env=env,
155                )
156                self._read_cm, self._write_cm = await stdio_client(
157                    server_params
158                ).__aenter__()
159                self._session = ClientSession(self._read_cm, self._write_cm)
160
161            elif self._config.transport == "sse":
162                if not self._config.url:
163                    raise HarnessError(
164                        f"MCP server '{self._config.name}' configured as sse "
165                        "but no url was provided.",
166                        failure_class=FailureClass.MCP_CONNECT_ERROR,
167                    )
168                try:
169                    from mcp.client.sse import sse_client
170                except ImportError:
171                    from mcp.client.http import http_client as sse_client  # type: ignore[no-redef]
172
173                self._read_cm, self._write_cm = await sse_client(
174                    self._config.url
175                ).__aenter__()
176                self._session = ClientSession(self._read_cm, self._write_cm)
177            else:
178                raise HarnessError(
179                    f"Unsupported transport: {self._config.transport}",
180                    failure_class=FailureClass.MCP_CONNECT_ERROR,
181                )
182
183            # Initialize the MCP session
184            await self._session.__aenter__()
185            await self._session.initialize()
186
187            # List tools
188            tools_result = await self._session.list_tools()

193                    description=mcp_tool.description or "",
194                    input_schema=mcp_tool.inputSchema or {"type": "object", "properties": {}},

196                    timeout_seconds=self._config.timeout,
