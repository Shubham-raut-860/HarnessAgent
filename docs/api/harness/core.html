229    def is_budget_ok(self) -> bool:
230        """Return True if the run has not exceeded any budget limit."""
231        if self.step_count >= self.max_steps:
232            return False
233        if self.token_count >= self.max_tokens:

235        if self.elapsed_seconds >= self.timeout_seconds:
236            return False
237        return True

239    def tick(self, tokens: int = 0) -> None:
240        """Advance step counter and accumulate token usage; raise on budget exceeded."""
241        self.step_count += 1
242        self.token_count += tokens
243
244        if self.step_count > self.max_steps:
245            self.failed = True
246            self.failure_class = FailureClass.BUDGET_STEPS.value
247            raise BudgetExceeded(
248                f"Step budget exceeded: {self.step_count}/{self.max_steps}",
249                failure_class=FailureClass.BUDGET_STEPS,
250                context={"run_id": self.run_id, "step_count": self.step_count},
251            )
252        if self.token_count > self.max_tokens:
253            self.failed = True
254            self.failure_class = FailureClass.BUDGET_TOKENS.value
255            raise BudgetExceeded(
256                f"Token budget exceeded: {self.token_count}/{self.max_tokens}",
257                failure_class=FailureClass.BUDGET_TOKENS,
258                context={"run_id": self.run_id, "token_count": self.token_count},
259            )
260        if self.elapsed_seconds > self.timeout_seconds:
261            self.failed = True
262            self.failure_class = FailureClass.BUDGET_TIME.value
263            raise BudgetExceeded(
264                f"Time budget exceeded: {self.elapsed_seconds:.1f}s / {self.timeout_seconds}s",
265                failure_class=FailureClass.BUDGET_TIME,
266                context={"run_id": self.run_id, "elapsed_seconds": self.elapsed_seconds},
267            )
