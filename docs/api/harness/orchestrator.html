163    async def shutdown(self, drain_timeout: float = 30.0) -> None:
164        """Set shutdown flag and wait up to drain_timeout for in-flight runs.
165
166        Any run still active after the timeout is marked 'failed' in Redis
167        with error_message='Run interrupted by graceful shutdown'.
168
169        Args:
170            drain_timeout: Seconds to wait before force-marking stragglers.
171        """
172        self._request_shutdown()
173
174        if not self._inflight:
175            logger.info("AgentRunner: no in-flight runs — clean shutdown")
176            return
177
178        logger.info(
179            "AgentRunner: draining %d in-flight run(s) (timeout=%.1fs)...",
180            len(self._inflight),
181            drain_timeout,
182        )
183        loop = asyncio.get_event_loop()
184        deadline = loop.time() + drain_timeout
185        while self._inflight:
186            remaining = deadline - loop.time()
187            if remaining <= 0:
188                break
189            await asyncio.sleep(min(1.0, remaining))
190
191        if self._inflight:
192            logger.warning(
193                "AgentRunner: drain timeout — interrupting %d run(s): %s",
194                len(self._inflight),
195                list(self._inflight),
196            )
197            for run_id in list(self._inflight):
198                try:
199                    record = await self.get_run(run_id)
200                    if record and record.status == "running":
201                        record.status = "failed"
202                        record.result = {
203                            "run_id": run_id,
204                            "output": "",
205                            "steps": 0,
206                            "tokens": 0,
207                            "success": False,
208                            "error_message": "Run interrupted by graceful shutdown",
209                        }
210                        record.completed_at = datetime.now(UTC)
211                        await self.update_run(record)
212                except Exception as exc:
213                    logger.debug("Could not mark run %s interrupted: %s", run_id, exc)
214            self._inflight.clear()
215        else:
216            logger.info("AgentRunner: all in-flight runs completed — clean shutdown")

299
300        Args:
301            run_id: The run to execute.
302
303        Returns:
304            The updated RunRecord.
305        """
306        record = await self.get_run(run_id)
307        if record is None:
308            raise KeyError(f"Run not found: {run_id}")
309        if record.status == "cancelled":
310            logger.info("Skipping cancelled run %s", run_id)
311            return record
312
313        # Reject new runs during shutdown
314        if self._shutting_down:
315            logger.warning("Rejecting run %s — shutdown in progress", run_id)
316            record.status = "cancelled"
317            record.completed_at = datetime.now(UTC)
318            await self.update_run(record)
319            return record
320
321        # Mark as running and track as in-flight
322        record.status = "running"
323        record.started_at = datetime.now(UTC)
324        await self.update_run(record)
325        self._inflight.add(run_id)
326
327        workspace = self._workspace_base / record.tenant_id / record.run_id
328        workspace.mkdir(parents=True, exist_ok=True)
329
330        try:
331            agent = self._agent_factory(record.agent_type)
332            agent_result = await _run_agent(agent, record, workspace)
333
334            record.status = "completed" if getattr(agent_result, "success", False) else "failed"
335            record.result = _serialise_result(agent_result)
336
337        except Exception as exc:
338            logger.exception("Run %s raised unhandled exception: %s", run_id, exc)
339            record.status = "failed"
340            record.result = {
341                "run_id": run_id,
342                "output": "",
343                "steps": 0,
344                "tokens": 0,
345                "success": False,
346                "error_message": str(exc),
347            }
348            if self._error_collector is not None:
349                try:
350                    await self._error_collector.record(
351                        agent_type=record.agent_type,
352                        task=record.task,
353                        failure_class="UNKNOWN",
354                        error_message=str(exc),
355                    )
356                except Exception:
357                    pass
358
359        finally:
360            self._inflight.discard(run_id)
361
362        record.completed_at = datetime.now(UTC)
363        await self.update_run(record)
364
365        logger.info(
366            "Run %s finished with status=%s", run_id, record.status
367        )

155                            for st in ready

169                            results[st.id] = AgentResult(
170                                run_id=f"failed_{st.id}",
171                                output="",
172                                steps=0,
173                                tokens=0,
174                                success=False,
175                                failure_class="UNKNOWN",
176                                error_message=str(result),
177                            )
178                            failed_ids.add(st.id)
179                        else:
180                            results[st.id] = result
181                            if not getattr(result, "success", True):
182                                failed_ids.add(st.id)
183
184                        completed_ids.add(st.id)
185                        await self._publish(
186                            "subtask_completed",
187                            plan.plan_id,
188                            {
189                                "subtask_id": st.id,
190                                "success": (
191                                    results[st.id].success
192                                    if hasattr(results[st.id], "success")
193                                    else False
194                                ),
195                            },
196                        )
197
198        except TimeoutError:
199            logger.error("Plan %s timed out after %.1f seconds", plan.plan_id, timeout)
200            await self._publish(
201                "plan_timeout", plan.plan_id, {"completed": list(completed_ids)}
202            )
203            raise
204
205        succeeded = sum(1 for r in results.values() if getattr(r, "success", False))
206        logger.info(
207            "Plan %s completed: %d/%d tasks succeeded",
208            plan.plan_id,
209            succeeded,
210            len(results),
211        )
212        await self._publish(
213            "plan_completed",
214            plan.plan_id,
215            {"total": len(results), "succeeded": succeeded},
216        )
217
218        return results

Algorithm:

  1. Validate the DAG and compute topological order.
  2. Maintain a set of completed task ids and their results.
  3. In each iteration, find all tasks ready to run (deps satisfied), execute them in parallel with asyncio.gather, then add to completed.
  4. Repeat until all tasks are done or timeout.

Args: plan: The TaskPlan to execute. tenant_id: The tenant executing this plan. timeout: Total wall-clock timeout for the entire plan (seconds).

Returns: Dict mapping subtask_id to AgentResult.

 78

 97                in_degree[neighbour_id] -= 1
