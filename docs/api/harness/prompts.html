143    async def apply_patch(self, patch: Any) -> PromptVersion:
144        """Apply a Hermes patch to the current active prompt.
145
146        Supported patch operations (patch.op):
147            - ``append``:  Append patch.value to the end of the prompt.
148            - ``prepend``: Prepend patch.value to the beginning.
149            - ``replace``: Replace patch.path substring with patch.value.
150            - ``remove``:  Remove patch.path substring from the prompt.
151            - ``set``:     Replace the entire prompt with patch.value.
152
153        Creates a new PromptVersion and promotes it.  Clears the cache.
154
155        Args:
156            patch: A Patch dataclass (harness.improvement.patch_generator).
157
158        Returns:
159            The newly created and promoted PromptVersion.
160
161        Raises:
162            ValueError: If the patch operation is unknown.
163        """
164        agent_type = getattr(patch, "agent_type", "")
165        op = getattr(patch, "op", "").lower()
166        path = getattr(patch, "path", "") or ""
167        value = getattr(patch, "value", "") or ""
168        patch_id = getattr(patch, "patch_id", None)
169
170        # Get current content
171        current_content = await self.get_prompt(agent_type)
172
173        # Apply the operation
174        if op == "append":
175            new_content = current_content.rstrip() + "\n\n" + value.strip()
176        elif op == "prepend":
177            new_content = value.strip() + "\n\n" + current_content.lstrip()
178        elif op == "replace":
179            if not path:
180                raise ValueError("Patch op='replace' requires a non-empty path (substring to replace)")
181            new_content = current_content.replace(path, value, 1)
182        elif op == "remove":
183            if not path:
184                raise ValueError("Patch op='remove' requires a non-empty path (substring to remove)")
185            new_content = current_content.replace(path, "", 1)
186        elif op == "set":
187            new_content = value
188        else:
189            raise ValueError(
190                f"Unknown patch operation: {op!r}. "
191                "Supported: append, prepend, replace, remove, set"
192            )
193
194        # Create new version
195        new_version = await self._store.create_version(
196            agent_type=agent_type,
197            content=new_content,
198            patch_id=patch_id,
199            created_by="hermes",
200            tags=["auto-generated", f"patch:{patch_id}" if patch_id else "patch"],
201            metadata={
202                "patch_op": op,
203                "patch_rationale": getattr(patch, "rationale", ""),
204                "based_on_errors": getattr(patch, "based_on_errors", []),
205            },
206        )
207
208        # Promote new version (returns the updated object with active=True) and clear cache
209        promoted = await self._store.promote(new_version.version_id)
210        self._clear_cache(agent_type)
211
212        logger.info(
213            "Applied patch %s to agent_type=%s (new version_id=%s)",
214            patch_id,
215            agent_type,
216            promoted.version_id[:8],
217        )
218        return promoted

class PromptStore: 
 31class PromptStore:
 32    """Versioned prompt storage in Redis.
 33
 34    Each prompt version is stored as a JSON hash at:
 35        harness:prompt:{agent_type}:{version_id}
 36
 37    An ordered index per agent_type is maintained as a sorted set at:
 38        harness:prompt_index:{agent_type}
 39    scored by version_number, so ZREVRANGE retrieves newest first.
 40
 41    The active version ID per agent_type is stored as a plain string at:
 42        harness:prompt_active:{agent_type}
 43
 44    Args:
 45        redis: An aioredis-compatible async Redis client.
 46    """
 47
 48    def __init__(self, redis) -> None:
 49        self._redis = redis
 50

 55    async def save(self, version: PromptVersion) -> PromptVersion:
 56        """Persist a PromptVersion to Redis.
 57
 58        Stores the JSON blob and updates the sorted index for the agent_type.
 59
 60        Args:
 61            version: The PromptVersion to save.
 62
 63        Returns:
 64            The saved PromptVersion (possibly with updated version_id).
 65        """
 66        key = _version_key(version.agent_type, version.version_id)
 67        index_key = _index_key(version.agent_type)
 68
 69        async with self._redis.pipeline(transaction=True) as pipe:
 70            pipe.set(key, version.to_json())
 71            pipe.zadd(index_key, {version.version_id: float(version.version_number)})
 72            if version.active:
 73                pipe.set(_active_key(version.agent_type), version.version_id)
 74            await pipe.execute()
 75
 76        logger.debug(
 77            "Saved prompt version %s for agent_type=%s (v%d, active=%s)",
 78            version.version_id[:8],
 79            version.agent_type,
 80            version.version_number,
 81            version.active,
 82        )
 83        return version

 85    async def create_version(
 86        self,
 87        agent_type: str,
 88        content: str,
 89        **kwargs,
 90    ) -> PromptVersion:
 91        """Create a new PromptVersion with auto-incremented version_number.
 92
 93        Args:
 94            agent_type: The agent type the prompt belongs to.
 95            content:    The prompt text.
 96            **kwargs:   Extra keyword arguments forwarded to PromptVersion().
 97
 98        Returns:
 99            The newly created PromptVersion (saved to Redis).
100        """
101        # Determine next version number
102        index_key = _index_key(agent_type)
103        count = await self._redis.zcard(index_key)
104        next_version = int(count) + 1
105
106        version = PromptVersion(
107            agent_type=agent_type,
108            content=content,
109            version_number=next_version,
110            **kwargs,
111        )
112        await self.save(version)
113        return version

119    async def get(self, version_id: str, agent_type: str = "") -> Optional[PromptVersion]:
120        """Retrieve a PromptVersion by its version_id.
121
122        If agent_type is not provided, scans known agent types.
123
124        Args:
125            version_id: The UUID of the version to retrieve.
126            agent_type: Optional hint to narrow the key space.
127
128        Returns:
129            PromptVersion if found, else None.
130        """
131        if agent_type:
132            raw = await self._redis.get(_version_key(agent_type, version_id))
133            if raw:
134                return PromptVersion.from_json(raw if isinstance(raw, str) else raw.decode())
135            return None
136
137        # Scan for the version across all known agent types
138        pattern = f"{_KEY_PREFIX}:*:{version_id}"
139        keys = []
140        async for key in self._redis.scan_iter(match=pattern, count=100):
141            keys.append(key)
142
143        for key in keys:
144            raw = await self._redis.get(key)
145            if raw:
146                return PromptVersion.from_json(raw if isinstance(raw, str) else raw.decode())
147        return None

149    async def get_active(self, agent_type: str) -> Optional[PromptVersion]:
150        """Return the currently active PromptVersion for agent_type, or None.
151
152        Args:
153            agent_type: The agent type to look up.
154
155        Returns:
156            The active PromptVersion, or None if none is set.
157        """
158        active_id_raw = await self._redis.get(_active_key(agent_type))
159        if not active_id_raw:
160            return None
161        active_id = active_id_raw if isinstance(active_id_raw, str) else active_id_raw.decode()
162        return await self.get(version_id=active_id, agent_type=agent_type)

164    async def list_versions(
165        self, agent_type: str, limit: int = 20
166    ) -> list[PromptVersion]:
167        """Return up to *limit* versions for *agent_type*, newest first.
168
169        Args:
170            agent_type: The agent type to list versions for.
171            limit:      Maximum number of versions to return.
172
173        Returns:
174            List of PromptVersion objects ordered by version_number descending.
175        """
176        index_key = _index_key(agent_type)
177        # ZREVRANGE returns members with highest scores first (highest version_number)
178        version_ids: list = await self._redis.zrevrange(
179            index_key, 0, limit - 1
180        )
181
182        versions: list[PromptVersion] = []
183        for vid_raw in version_ids:
184            vid = vid_raw if isinstance(vid_raw, str) else vid_raw.decode()
185            v = await self.get(version_id=vid, agent_type=agent_type)
186            if v is not None:
187                versions.append(v)
188
189        return versions

195    async def promote(self, version_id: str) -> PromptVersion:
196        """Make the specified version the active one for its agent_type.
197
198        Deactivates all other versions for the same agent_type.
199
200        Args:
201            version_id: The version to promote.
202
203        Returns:
204            The promoted PromptVersion.
205
206        Raises:
207            KeyError: If version_id is not found.
208        """
209        # Fetch the target version (scan since we don't always know agent_type)
210        target = await self.get(version_id=version_id)
211        if target is None:
212            raise KeyError(f"PromptVersion not found: {version_id}")
213
214        agent_type = target.agent_type
215
216        # Deactivate all existing versions for this agent_type
217        all_versions = await self.list_versions(agent_type, limit=1000)
218        async with self._redis.pipeline(transaction=True) as pipe:
219            for v in all_versions:
220                if v.version_id != version_id and v.active:
221                    v.active = False
222                    pipe.set(_version_key(agent_type, v.version_id), v.to_json())
223            # Activate target
224            target.active = True
225            pipe.set(_version_key(agent_type, version_id), target.to_json())
226            pipe.set(_active_key(agent_type), version_id)
