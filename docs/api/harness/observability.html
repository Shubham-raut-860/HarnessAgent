211        self,
212        run_id: str,
213        task: str,
214        output: str,
215        expected: str | None = None,
216    ) -> dict[str, Any]:
217        """
218        Log evaluation metrics for a completed agent run.
219
220        Always computes:
221        - response_length: character count of the output.
222        - step_count: taken from active run metrics if available.
223
224        If ``expected`` is provided, additional quality metrics are logged.
225        """
226        if not _MLFLOW_AVAILABLE:
227            return {}
228
229        mlflow_run_id = self._active_run_ids.get(run_id)
230
231        metrics: dict[str, float] = {
232            "eval_response_length": float(len(output)),
233            "eval_task_length": float(len(task)),
234        }
235
236        if expected:
237            # Simple character-level overlap as a proxy for quality
238            overlap = len(set(output.lower().split()) & set(expected.lower().split()))
239            total = len(set(expected.lower().split())) or 1
240            metrics["eval_word_overlap"] = float(overlap / total)
241
242        try:
243            if mlflow_run_id:
244                with mlflow.start_run(run_id=mlflow_run_id):
245                    mlflow.log_metrics(metrics)
246            else:
247                # No active run — log to a new evaluation run
248                with mlflow.start_run(run_name=f"eval_{run_id[:8]}"):
249                    mlflow.log_metrics(metrics)
250                    mlflow.log_params({"task": task[:256], "run_id": run_id})
251        except Exception as exc:
252            logger.warning("MLflow evaluate_run failed: %s", exc)
253
254        return metrics

 75        """
 76        Async generator that yields StepEvents published for ``run_id``.
 77
 78        Automatically unsubscribes when the generator is closed (GeneratorExit).
 79        Each subscriber uses its own dedicated Redis connection.
 80        """
 81        channel = f"{_CHANNEL_PREFIX}{run_id}"
 82        client = await self._create_subscriber()
 83        pubsub = client.pubsub()
 84
 85        try:
 86            await pubsub.subscribe(channel)
 87            logger.debug("EventBus: subscribed to channel %s", channel)
 88
 89            async for message in pubsub.listen():
 90                if message is None:
 91                    continue
 92
 93                msg_type = message.get("type")
 94                if msg_type != "message":
 95                    # Skip subscribe/unsubscribe confirmation messages
 96                    continue
 97
 98                data = message.get("data")
 99                if not data:
100                    continue

102                try:
103                    payload = json.loads(data)
104                    ts_raw = payload.get("timestamp")

106                        datetime.fromisoformat(ts_raw)
107                        if ts_raw
108                        else datetime.now(timezone.utc)
109                    )
110                    step_event = StepEvent(
111                        run_id=payload.get("run_id", run_id),
112                        step=int(payload.get("step", 0)),
113                        event_type=payload.get("event_type", "unknown"),
114                        payload=payload.get("payload", {}),
115                        timestamp=ts,
116                    )
117                    yield step_event
118                except (json.JSONDecodeError, KeyError, ValueError, TypeError) as exc:
119                    logger.debug("EventBus: malformed message skipped: %s", exc)
120                    continue
121
122        except GeneratorExit:
123            logger.debug("EventBus: subscriber for run %s closed", run_id)
124        finally:
125            try:
126                await pubsub.unsubscribe(channel)
127                await pubsub.aclose()
128                await client.aclose()
129                if client in self._sub_clients:
130                    self._sub_clients.remove(client)
131            except Exception as exc:
