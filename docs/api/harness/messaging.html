 23class AgentMessageBus:
 24    """
 25    Redis Streams-backed message bus for inter-agent communication.
 26
 27    Each agent has a dedicated stream: ``harness:stream:{agent_id}``
 28    A shared broadcast stream: ``harness:stream:broadcast``
 29
 30    Message routing:
 31    - ``send(msg)``          → XADD to recipient or broadcast stream
 32    - ``subscribe(agent_id)``→ XREAD blocking on agent + broadcast streams
 33    - ``request(...)``       → send + await correlated reply
 34    - ``fan_out(...)``       → send to N agents, gather replies
 35    """

 60    async def send(self, msg: AgentMessage) -> str:
 61        """
 62        Publish a message to the appropriate Redis stream.
 63
 64        Returns the stream entry ID assigned by Redis.
 65        Also registers the message in the TTL index.
 66        """
 67        client = await self._get_client()
 68        stream_key = (
 69            _BROADCAST_STREAM
 70            if msg.is_broadcast()
 71            else f"{_STREAM_PREFIX}{msg.recipient_id}"
 72        )
 73        serialized = json.dumps(msg.to_dict(), default=str)
 74
 75        try:
 76            entry_id: str = await client.xadd(
 77                stream_key,
 78                {"data": serialized},
 79                maxlen=10_000,
 80                approximate=True,
 81            )
 82
 83            # Register in TTL index (score = expire timestamp)
 84            expire_ts = time.time() + msg.ttl_seconds
 85            await client.zadd(_MSG_INDEX_KEY, {msg.id: expire_ts})
 86
 87            return entry_id
 88        except Exception as exc:
 89            logger.error("AgentMessageBus.send failed: %s", exc)
 90            raise

 92    async def subscribe(
 93        self,
 94        agent_id: str,
 95        message_types: list[str] | None = None,
 96        last_id: str = "$",
 97    ) -> AsyncIterator[AgentMessage]:
 98        """
 99        Async generator yielding messages for ``agent_id``.
100
101        Reads from both:
102        - ``harness:stream:{agent_id}`` (direct messages)
103        - ``harness:stream:broadcast`` (broadcast messages)
104
105        Filters by ``message_types`` if specified.
106        Expired messages are silently skipped.
107        Uses XREAD with BLOCK=100ms to avoid busy-waiting.
108        """
109        client = await self._get_client()
110        agent_stream = f"{_STREAM_PREFIX}{agent_id}"
111
112        # Maintain per-stream cursor IDs
113        cursors: dict[str, str] = {
114            agent_stream: last_id,
115            _BROADCAST_STREAM: last_id,
116        }
117
118        try:
119            while True:
120                streams_to_read = list(cursors.keys())
121                ids_to_read = list(cursors.values())
122
123                try:
124                    results = await client.xread(
125                        streams={k: v for k, v in zip(streams_to_read, ids_to_read)},
126                        block=_BLOCK_MS,
127                        count=50,
128                    )
129                except asyncio.CancelledError:
130                    break
131                except Exception as exc:
132                    logger.warning("XREAD failed: %s — retrying", exc)
133                    await asyncio.sleep(0.5)
134                    continue
135
136                if not results:
137                    continue
138
139                for stream_name, entries in results:
140                    # Update cursor
141                    if entries:
142                        cursors[stream_name] = entries[-1][0]
143
144                    for entry_id, data in entries:
145                        raw = data.get("data")
146                        if not raw:
147                            continue
148                        try:
149                            msg_dict = json.loads(raw)
150                            msg = AgentMessage.from_dict(msg_dict)
151                        except (json.JSONDecodeError, KeyError, TypeError) as exc:
152                            logger.debug("Message deserialisation failed: %s", exc)
153                            continue
154
155                        # Skip expired messages
156                        if msg.is_expired():
157                            continue
158
159                        # Filter by message type
160                        if message_types and msg.message_type not in message_types:
161                            continue
162
163                        yield msg
164
165        except GeneratorExit:
166            pass

168    async def request(
169        self,
170        sender_id: str,
171        recipient_id: str,
172        payload: dict[str, Any],
173        timeout: float = 30.0,
174    ) -> AgentMessage:
175        """
176        Send a query and await a correlated result or error reply.
177
178        Raises InterAgentTimeout if no reply arrives within ``timeout`` seconds.
179        """
180        import uuid as _uuid
181
182        correlation_id = _uuid.uuid4().hex
183        query_msg = AgentMessage(
184            sender_id=sender_id,
185            recipient_id=recipient_id,
186            message_type="query",
187            payload=payload,
188            correlation_id=correlation_id,
189        )
190        await self.send(query_msg)
191
192        deadline = asyncio.get_running_loop().time() + timeout
193        async for reply in self.subscribe(
194            sender_id, message_types=["result", "error"]
195        ):
196            if reply.correlation_id == correlation_id:
197                return reply
198            if asyncio.get_running_loop().time() > deadline:
199                break
200
201        raise InterAgentTimeout(
202            f"No reply from {recipient_id} within {timeout}s",
203            context={"sender": sender_id, "correlation_id": correlation_id},
204        )

206    async def fan_out(
207        self,
208        sender_id: str,
209        recipient_ids: list[str],
210        payload: dict[str, Any],
211        timeout: float = 60.0,
212    ) -> list[AgentMessage]:
213        """
214        Send the same payload to multiple recipients and collect replies.
215
216        Returns partial results — if some recipients time out their replies
217        are simply omitted from the list.
218        """
219        import uuid as _uuid
220
221        correlation_id = _uuid.uuid4().hex
222        pending: set[str] = set(recipient_ids)
223        replies: list[AgentMessage] = []
224
225        # Send to all recipients in parallel
226        send_tasks = [
227            self.send(
228                AgentMessage(
229                    sender_id=sender_id,
230                    recipient_id=rid,
231                    message_type="task",
232                    payload=payload,
233                    correlation_id=correlation_id,
234                )
235            )
236            for rid in recipient_ids
237        ]
238        await asyncio.gather(*send_tasks, return_exceptions=True)
239
240        # Collect replies
241        deadline = asyncio.get_running_loop().time() + timeout
242        async for reply in self.subscribe(
243            sender_id, message_types=["result", "error"]
244        ):
245            if reply.correlation_id == correlation_id and reply.sender_id in pending:
246                replies.append(reply)
247                pending.discard(reply.sender_id)
248                if not pending:
249                    break
250            if asyncio.get_running_loop().time() > deadline:
251                break
252
253        return replies
254
255    async def broadcast(
256        self,
257        sender_id: str,
258        message_type: str,
259        payload: dict[str, Any],
260    ) -> str:
261        """Publish a broadcast message to all subscribers."""
262        msg = AgentMessage(
263            sender_id=sender_id,
264            recipient_id=None,
265            message_type=message_type,  # type: ignore[arg-type]
266            payload=payload,
267        )
268        return await self.send(msg)

270    async def cleanup_expired(self) -> int:
271        """
272        Remove expired message IDs from the TTL index.
273
274        Returns the number of entries removed.
275        """
276        client = await self._get_client()
277        try:
278            now = time.time()
279            removed: int = await client.zremrangebyscore(

284            return removed

287            return 0

 18class MessagePatterns:
 19    """
 20    High-level messaging patterns:
 21    - pipeline    : linear chain of agents
 22    - scatter_gather: fan-out + selective collect
 23    - heartbeat_monitor: periodic liveness checking
 24    """
 25
 26    def __init__(self, bus: AgentMessageBus) -> None:
 27        self._bus = bus
 28
 29    async def pipeline(
 30        self,
 31        agents: list[str],
 32        initial_payload: dict[str, Any],
 33        timeout: float = 60.0,
 34        sender_id: str = "orchestrator",
 35    ) -> AgentMessage:
 36        """
 37        Chain agents in sequence: output of agent[i] becomes input of agent[i+1].
 38
 39        Returns the final AgentMessage from the last agent.
 40        Raises InterAgentTimeout if any agent in the chain doesn't respond.
 41        """
 42        if not agents:
 43            raise ValueError("agents list must not be empty")
 44
 45        current_payload = initial_payload
 46
 47        for i, agent_id in enumerate(agents):
 48            logger.debug(
 49                "Pipeline step %d/%d: sending to %s", i + 1, len(agents), agent_id
 50            )
 51            try:
 52                reply = await self._bus.request(
 53                    sender_id=sender_id,
 54                    recipient_id=agent_id,
 55                    payload=current_payload,
 56                    timeout=timeout,
 57                )
 58                if reply.message_type == "error":
 59                    logger.warning(
 60                        "Pipeline error at step %d (agent=%s): %s",
 61                        i + 1,
 62                        agent_id,
 63                        reply.payload,
 64                    )
 65                    return reply
 66                current_payload = reply.payload
 67            except InterAgentTimeout:
 68                logger.error(
 69                    "Pipeline timeout at step %d (agent=%s)", i + 1, agent_id
 70                )
 71                raise
 72

144        self,
145        agent_ids: list[str],
146        interval: float = 30.0,
147        sender_id: str = "monitor",
148    ) -> AsyncIterator[dict[str, str]]:
149        """
150        Async generator that periodically pings each agent and yields a
151        health-status dict: {agent_id: "alive" | "dead" | "unknown"}.
152
153        Uses a fire-and-forget heartbeat pattern: sends a ``heartbeat`` message
154        and waits up to ``interval/2`` seconds for a reply.
155        """
156        while True:
157            status: dict[str, str] = {aid: "unknown" for aid in agent_ids}

161                for agent_id in agent_ids
162            ]

167                    status[agent_id] = "dead"
168                elif result:
169                    status[agent_id] = "alive"
170                else:
171                    status[agent_id] = "dead"
172
173            yield status
174            await asyncio.sleep(interval)

180    async def _ping_agent(
181        self, sender_id: str, agent_id: str, timeout: float
182    ) -> bool:
183        """Send a heartbeat and return True if the agent replies in time."""
184        try:
185            reply = await self._bus.request(
186                sender_id=sender_id,
187                recipient_id=agent_id,
188                payload={"type": "ping"},
189                timeout=timeout,
190            )
191            return reply.message_type in ("result", "status", "heartbeat")
192        except InterAgentTimeout:
193            return False
194        except Exception as exc:
195            logger.debug("Heartbeat ping to %s failed: %s", agent_id, exc)
196            return False

34        sender_id: str = "orchestrator",
35    ) -> AgentMessage:
36        """
37        Chain agents in sequence: output of agent[i] becomes input of agent[i+1].

57                )

60                        "Pipeline error at step %d (agent=%s): %s",
61                        i + 1,
62                        agent_id,
63                        reply.payload,
64                    )
65                    return reply

67            except InterAgentTimeout:

74        return AgentMessage(
75            sender_id=agents[-1],
76            recipient_id=sender_id,
77            message_type="result",
