 24class ShortTermMemory:
 25    """
 26    Redis-backed short-term memory for per-run conversation history and scratch pad.
 27
 28    Conversation messages are stored as JSON strings in a Redis List (LPUSH),
 29    so index 0 is always the most recent message (reversed order).
 30    The scratch pad uses a Redis Hash scoped to the run_id with a TTL.
 31    """
 32
 33    def __init__(self, redis_url: str) -> None:
 34        self._redis_url = redis_url
 35        self._pool: ConnectionPool | None = None
 36        self._client: aioredis.Redis | None = None
 37
 38    # ------------------------------------------------------------------
 39    # Connection management
 40    # ------------------------------------------------------------------
 41
 42    async def _get_client(self) -> aioredis.Redis:
 43        if self._client is None:
 44            self._pool = aioredis.ConnectionPool.from_url(
 45                self._redis_url,
 46                max_connections=20,
 47                decode_responses=True,
 48            )
 49            self._client = aioredis.Redis(connection_pool=self._pool)
 50        return self._client
 51
 52    async def close(self) -> None:
 53        """Release Redis connection pool."""
 54        if self._client is not None:
 55            await self._client.aclose()
 56            self._client = None
 57        if self._pool is not None:
 58            await self._pool.aclose()
 59            self._pool = None
 60
 61    # ------------------------------------------------------------------
 62    # Internal helpers
 63    # ------------------------------------------------------------------
 64
 65    def _conv_key(self, run_id: str) -> str:
 66        return f"{_CONV_PREFIX}{run_id}"
 67
 68    def _scratch_key(self, run_id: str) -> str:
 69        return f"{_SCRATCH_PREFIX}{run_id}"
 70
 71    def _serialize_message(self, msg: ConversationMessage) -> str:
 72        return json.dumps(
 73            {
 74                "role": msg.role,
 75                "content": msg.content,
 76                "timestamp": msg.timestamp.isoformat(),
 77                "tokens": msg.tokens,
 78            }
 79        )
 80
 81    def _deserialize_message(self, raw: str) -> ConversationMessage:
 82        data = json.loads(raw)
 83        ts_raw = data.get("timestamp")
 84        ts = (
 85            datetime.fromisoformat(ts_raw)
 86            if ts_raw
 87            else datetime.now(timezone.utc)
 88        )
 89        return ConversationMessage(
 90            role=data["role"],
 91            content=data["content"],
 92            timestamp=ts,
 93            tokens=data.get("tokens", 0),
 94        )
 95
 96    async def _safe_exec(self, coro: Any) -> Any:
 97        """Execute a Redis coroutine and translate connection errors to HarnessError."""
 98        try:
 99            return await coro
100        except RedisConnectionError as exc:
101            logger.warning("Redis connection error: %s", exc)
102            raise HarnessError(
103                f"Redis unavailable: {exc}",
104                failure_class=FailureClass.MEMORY_REDIS,
105                context={"redis_url": self._redis_url},
106            ) from exc
107

109    # Public API

112    async def push_message(
113        self,
114        run_id: str,
115        role: str,
116        content: str,
117        tokens: int = 0,
118    ) -> None:
119        """Prepend a new conversation message to the Redis list for run_id."""
120        client = await self._get_client()
121        msg = ConversationMessage(
122            role=role,  # type: ignore[arg-type]
123            content=content,
124            tokens=tokens,
125        )
126        serialized = self._serialize_message(msg)
127        await self._safe_exec(client.lpush(self._conv_key(run_id), serialized))
128
129    async def get_history(
130        self,
131        run_id: str,
132        last_n: int = 20,
133    ) -> list[ConversationMessage]:
134        """
135        Retrieve the most recent ``last_n`` messages for this run.
136        Returns newest-first order is reversed to chronological order.
137        """
138        client = await self._get_client()
139        # Messages are LPUSH'd so index 0 = newest; LRANGE 0..last_n-1 gives newest→oldest
140        raw_items: list[str] = await self._safe_exec(
141            client.lrange(self._conv_key(run_id), 0, last_n - 1)
142        )
143        # Reverse so we return chronological order (oldest first)
144        messages = [self._deserialize_message(raw) for raw in reversed(raw_items)]
145        return messages
146
147    async def set_scratch(
148        self,
149        run_id: str,
150        key: str,
151        value: Any,
152        ttl: int = 3600,
153    ) -> None:
154        """Store an arbitrary JSON-serialisable value in the scratch hash."""
155        client = await self._get_client()
156        scratch_key = self._scratch_key(run_id)
157        serialized_value = json.dumps(value, default=str)
158        await self._safe_exec(client.hset(scratch_key, key, serialized_value))
159        await self._safe_exec(client.expire(scratch_key, ttl))
160
161    async def get_scratch(self, run_id: str, key: str) -> Any | None:
162        """Retrieve a value from the scratch hash; returns None if missing."""
163        client = await self._get_client()
164        raw = await self._safe_exec(
165            client.hget(self._scratch_key(run_id), key)
166        )
167        if raw is None:
168            return None
169        try:
170            return json.loads(raw)
171        except (json.JSONDecodeError, TypeError):
172            return raw
173
174    async def clear(self, run_id: str) -> None:
175        """Delete conversation list and scratch hash for the given run_id."""
176        client = await self._get_client()
177        await self._safe_exec(
178            client.delete(self._conv_key(run_id), self._scratch_key(run_id))
179        )
180
181    async def get_context_size(self, run_id: str) -> int:
182        """Return the number of messages stored for this run."""
183        client = await self._get_client()
184        length: int = await self._safe_exec(
185            client.llen(self._conv_key(run_id))
186        )
187        return length

 85        self,
 86        query: str,
 87        ctx: "AgentContext",
 88        max_hops: int = 2,
 89        vector_k: int = 5,
 90    ) -> RetrievalResult:
 91        """Retrieve context using weighted graph traversal with vector bridging."""
 92        entities = await extract_entities(query, llm_provider=self._llm)
 93        graph_paths: list[GraphPath] = []
 94        graph_context = ""
 95        anchor_source = "none"
 96
 97        # Step 1: find anchor nodes from entity names
 98        anchor_ids: list[str] = []
 99        if entities:
100            try:
101                anchor_nodes = await self._graph.find_nodes(entities, fuzzy=True)
102                anchor_ids = [n.id for n in anchor_nodes]
103                anchor_source = "regex"
104            except Exception as exc:
105                logger.warning("Graph anchor lookup failed: %s", exc)
106
107        # Step 2: vector-to-graph bridging if regex found nothing
108        vector_hits = []
109        try:
110            tenant_filter = {"tenant_id": ctx.tenant_id} if ctx.tenant_id else None
111            vector_hits = await self._vector_store.query(
112                text=query, k=vector_k, filter=tenant_filter
113            )
114        except Exception as exc:
115            logger.warning("Vector retrieval failed: %s", exc)
116
117        if not anchor_ids and vector_hits:
118            # Pull entity-like words from the top vector results and try those as anchors
119            bridge_entities: list[str] = []
120            for hit in vector_hits[:3]:
121                bridge_entities.extend(await extract_entities(hit.text))
122            if bridge_entities:
123                try:
124                    bridge_nodes = await self._graph.find_nodes(bridge_entities, fuzzy=True)
125                    anchor_ids = [n.id for n in bridge_nodes]
126                    anchor_source = "vector_bridge"
127                except Exception as exc:
128                    logger.warning("Vector bridge anchor lookup failed: %s", exc)
129
130        # Step 3: weighted BFS traversal from all anchors simultaneously
131        if anchor_ids:
132            try:
133                raw_paths = await self._graph.traverse(anchor_ids, max_hops=max_hops)
134                scored = self._score_paths(raw_paths)
135                # Take top paths by score
136                top_paths = [sp.path for sp in scored[:self._max_paths]]
137                graph_paths = top_paths
138                # Update frequency counters (lock guards concurrent coroutine writes)
139                async with self._freq_lock:
140                    for path in graph_paths:
141                        for node in path.nodes:
142                            self._node_freq[node.id] += 1
143                graph_context = self._render_paths(graph_paths)
144            except Exception as exc:

148
149        # Strategy annotation

151            strategy = "graph_primary"

154        elif anchor_source == "vector_bridge":

163            graph_context=graph_context,

165            vector_context=vector_context,
166            total_tokens_estimate=self._estimate_tokens(

291    def _score_paths(self, paths: list[GraphPath]) -> list[ScoredPath]:
292        """Score each path by cumulative edge weight + node frequency bonus."""
293        scored: list[ScoredPath] = []
294        for path in paths:
295            edge_score = sum(
296                p.props.get("weight", _DEFAULT_EDGE_WEIGHT)
297                for p in path.edges
298            ) if path.edges else _DEFAULT_EDGE_WEIGHT
299
300            # Frequency bonus: nodes accessed more often rank higher
301            freq_bonus = sum(
302                min(self._node_freq.get(n.id, 0) * 0.1, 1.0)
303                for n in path.nodes
304            )
305            scored.append(ScoredPath(path=path, score=edge_score + freq_bonus))
306
307        scored.sort(key=lambda s: s.score, reverse=True)
308        return scored

310    def _render_paths(self, paths: list[GraphPath]) -> str:
311        """Render paths as compact, deduplicated context text.
312
313        Sections:
314        [SCHEMA]    - table/column structure
315        [JOINS]     - join relationships
316        [HISTORY]   - past queries that touched these tables
317        [ERRORS]    - past failures (so agent avoids repeating them)
318        """
319        table_cols: dict[str, list[str]] = {}
320        join_lines: list[str] = []
321        query_lines: list[str] = []
322        error_lines: list[str] = []
323        seen: set[str] = set()
324
325        # Sort tables by frequency so most-used come first
326        for path in paths:
327            for node in path.nodes:
328                nid = node.id
329                if node.type == "Table" and nid not in table_cols:
330                    table_cols[nid] = []
331                elif node.type == "Column":
332                    table = node.props.get("table", "")
333                    col_name = node.props.get("name", nid)
334                    col_type = node.props.get("col_type", "?")
335                    nullable = "" if node.props.get("nullable", True) else " NOT NULL"
336                    key = f"{table}.{col_name}"
337                    if table and key not in seen:
338                        seen.add(key)
339                        table_cols.setdefault(table, []).append(
340                            f"{col_name}({col_type}{nullable})"
341                        )
342                elif node.type == "Query":
343                    sql = node.props.get("sql", "")
344                    success = node.props.get("success", True)
345                    latency = node.props.get("latency_ms")
346                    if sql and sql not in seen:
347                        seen.add(sql)
348                        lat_str = f", {latency:.0f}ms" if latency else ""
349                        status = "ok" if success else "FAILED"
350                        query_lines.append(f"  [{status}{lat_str}] {sql[:200]}")
351                elif node.type == "Error":
352                    msg = node.props.get("message", "")
353                    if msg and msg not in seen:
354                        seen.add(msg)
355                        error_lines.append(f"  {msg[:200]}")
356
357            for edge in path.edges:
358                if edge.type == "joins":
359                    on_clause = edge.props.get("on", "")
360                    line = f"{edge.source_id} --joins--> {edge.target_id}"
361                    if on_clause:
362                        line += f" ON {on_clause}"
363                    if line not in seen:
364                        seen.add(line)
365                        join_lines.append(line)
366
367        # Render — sort tables by frequency (descending)
368        sections: list[str] = []
369
370        if table_cols:
371            sections.append("[SCHEMA]")
372            for table_name in sorted(
373                table_cols,
374                key=lambda t: self._node_freq.get(t, 0),
375                reverse=True,
376            ):
377                cols = table_cols[table_name]
378                col_str = ", ".join(cols) if cols else "(no columns)"
379                freq = self._node_freq.get(table_name, 0)
380                freq_note = f" (used {freq}x)" if freq > 0 else ""
381                sections.append(f"{table_name}: Table | cols: {col_str}{freq_note}")
382
383        if join_lines:
384            sections.append("[JOINS]")
385            sections.extend(join_lines)
386
387        if query_lines:
388            sections.append("[PAST QUERIES]")
389            sections.extend(query_lines[:5])  # cap at 5 to keep context tight
390
391        if error_lines:
392            sections.append("[PAST ERRORS - avoid repeating]")
393            sections.extend(error_lines[:3])
394
395        return "\n".join(sections)
396
397    def _estimate_tokens(self, *texts: str) -> int:
398        return max(1, sum(len(t) for t in texts) // 4)

 18
 19    Responsibilities:
 20    - Track token budget across system prompt, retrieved context, and messages.
 21    - Implement a sliding window that keeps the most recent messages.
 22    - Optionally summarise dropped messages via an LLM call.
 23    """
 24
 25    def __init__(
 26        self,
 27        max_tokens: int,
 28        summarizer_provider: Any = None,
 29    ) -> None:
 30        self._max_tokens = max_tokens
 31        self._summarizer = summarizer_provider
 32
 33    # ------------------------------------------------------------------
 34    # Public API
 35    # ------------------------------------------------------------------
 36
 37    async def fit(
 38        self,
 39        messages: list[ConversationMessage],
 40        system_tokens: int = 0,
 41        retrieved_tokens: int = 0,
 42        reserve_output: int = 2000,
 43    ) -> ContextWindow:
 44        """
 45        Fit ``messages`` into the available token budget.
 46
 47        The available budget is:
 48            max_tokens - system_tokens - retrieved_tokens - reserve_output
 49
 50        System messages are always kept.  The most recent non-system messages
 51        are retained until the budget is consumed.  If any messages are
 52        dropped and a summarizer is configured, the dropped messages are
 53        compressed into a single summary message.
 54        """
 55        available = self._max_tokens - system_tokens - retrieved_tokens - reserve_output
 56        if available <= 0:
 57            logger.warning(
 58                "Context window fully consumed by system/retrieved tokens "
 59                "(available=%d); returning empty message list.",
 60                available,
 61            )
 62            return ContextWindow(
 63                messages=[],
 64                total_tokens=0,
 65                truncated=bool(messages),
 66                summary=None,
 67            )
 68
 69        # Separate system messages (always kept) from conversational messages
 70        system_msgs = [m for m in messages if m.role == "system"]
 71        conv_msgs = [m for m in messages if m.role != "system"]
 72
 73        system_budget_used = sum(self._msg_tokens(m) for m in system_msgs)
 74        remaining = available - system_budget_used
 75
 76        # Slide from the end (most recent) towards the front
 77        kept: list[ConversationMessage] = []
 78        tokens_used = 0
 79
 80        for msg in reversed(conv_msgs):
 81            t = self._msg_tokens(msg)
 82            if tokens_used + t <= remaining:
 83                kept.insert(0, msg)
 84                tokens_used += t
 85            else:
 86                # No more room
 87                break
 88
 89        dropped = conv_msgs[: len(conv_msgs) - len(kept)]
 90        truncated = bool(dropped)
 91        summary: str | None = None
 92
 93        if dropped and self._summarizer is not None:
 94            try:
 95                summary = await self.summarize(dropped)
 96                summary_msg = ConversationMessage(
 97                    role="system",
 98                    content=f"[Summary of earlier conversation]\n{summary}",
 99                    tokens=self._estimate_tokens(summary),
100                )
101                # Only include summary if it fits
102                summary_tokens = self._msg_tokens(summary_msg)
103                if tokens_used + summary_tokens <= remaining:
104                    kept.insert(0, summary_msg)
105                    tokens_used += summary_tokens
106            except Exception as exc:
107                logger.warning("Summarizer failed, continuing without summary: %s", exc)
108
109        final_messages = system_msgs + kept
110        total_tokens = system_budget_used + tokens_used
111
112        return ContextWindow(
113            messages=final_messages,
114            total_tokens=total_tokens,
115            truncated=truncated,
116            summary=summary,
117        )
118
119    async def summarize(self, messages: list[ConversationMessage]) -> str:
120        """
121        Compress a list of messages into a brief summary string.
122
123        When an LLM summarizer is available it is used; otherwise a rule-based
124        fallback extracts one user/assistant pair every 5 messages.
125        """
126        if self._summarizer is not None:
127            try:
128                formatted = "\n".join(
129                    f"{m.role.upper()}: {m.content[:500]}" for m in messages
130                )
131                prompt_messages = [
132                    {
133                        "role": "user",
134                        "content": (
135                            "Summarise the following conversation in 3-5 sentences, "
136                            "preserving key facts, decisions, and context:\n\n"
137                            + formatted
138                        ),
139                    }
140                ]
141                response = await self._summarizer.complete(
142                    messages=prompt_messages,
143                    max_tokens=400,
144                )
145                return response.content.strip()
146            except Exception as exc:
147                logger.warning("LLM summarizer call failed: %s", exc)
148
149        # Fallback: extract representative pairs
150        pairs: list[str] = []
151        for i in range(0, len(messages), 5):
152            chunk = messages[i : i + 5]
153            for m in chunk:
154                if m.role in ("user", "assistant"):
155                    snippet = m.content[:200].replace("\n", " ")
156                    pairs.append(f"{m.role}: {snippet}")
157                    break  # one per chunk
158
159        if not pairs:
160            return "(prior conversation omitted)"
161        return "Prior context: " + " | ".join(pairs)

163    def count_tokens(self, messages: list[ConversationMessage]) -> int:
164        """Total token count for a list of messages."""
165        return sum(self._msg_tokens(m) for m in messages)

173            return msg.tokens
