 75    def filter(
 76        self,
 77        tags: list[str] | None = None,
 78        n: int | None = None,
 79    ) -> EvalDataset:
 80        """Return a new EvalDataset containing only cases matching the given tags.
 81
 82        Args:
 83            tags: If provided, only include cases that have at least one
 84                  matching tag.  If None, all cases pass.
 85            n:    If provided, cap the result at n cases (preserves order).
 86
 87        Returns:
 88            A new EvalDataset with the filtered subset of cases.
 89        """
 90        filtered = self.cases
 91        if tags:
 92            tag_set = set(tags)
 93            filtered = [c for c in filtered if tag_set.intersection(c.tags)]
 94        if n is not None:
 95            filtered = filtered[:n]
 96        return EvalDataset(name=self.name, agent_type=self.agent_type, cases=filtered)

 99        """Return a new EvalDataset with n cases randomly sampled without replacement.

101        Args:
102            n: Number of cases to sample; clamped to len(self.cases).
103
104        Returns:
105            A new EvalDataset with the sampled cases in random order.
106        """
107        k = min(n, len(self.cases))
108        sampled = random.sample(self.cases, k)
109        return EvalDataset(name=self.name, agent_type=self.agent_type, cases=sampled)

116    def from_jsonl(cls, path: str) -> EvalDataset:
117        """Load a dataset from a JSONL file.
118
119        Each line must be a JSON object matching the EvalCase schema.
120        The first case's agent_type determines the dataset's agent_type.
121        The file stem is used as the dataset name.
122
123        Args:
124            path: Filesystem path to the .jsonl file.
125
126        Returns:
127            An EvalDataset populated from the file.
128
129        Raises:
130            FileNotFoundError: If the file does not exist.
131            json.JSONDecodeError: If any line is invalid JSON.
132        """
133        p = Path(path)
134        if not p.exists():
135            raise FileNotFoundError(f"EvalDataset file not found: {path}")
136
137        cases: list[EvalCase] = []
138        with p.open("r", encoding="utf-8") as fh:
139            for line_num, line in enumerate(fh, 1):
140                line = line.strip()
141                if not line:
142                    continue
143                try:
144                    data = json.loads(line)
145                    cases.append(EvalCase.from_dict(data))
146                except (json.JSONDecodeError, KeyError) as exc:
147                    raise ValueError(
148                        f"Invalid JSONL at {path}:{line_num}: {exc}"
149                    ) from exc
150
151        agent_type = cases[0].agent_type if cases else "unknown"
152        return cls(name=p.stem, agent_type=agent_type, cases=cases)

574        self,
575        tenant_id: str,
576        agent_type: str,
577        task: str,
578        metadata: dict[str, Any] | None = None,
579    ) -> Any:
580        """Invoke the agent runner using the most appropriate interface."""
581        metadata = metadata or {}
582
583        # Protocol 1: production AgentRunner lifecycle.
584        if hasattr(self._runner, "create_run") and hasattr(self._runner, "execute_run"):
585            record = await self._maybe_await(
586                self._runner.create_run(
587                    tenant_id=tenant_id,
588                    agent_type=agent_type,
589                    task=task,
590                    metadata=metadata,
591                )
592            )
593            updated = await self._maybe_await(self._runner.execute_run(record.run_id))
594            return _record_to_agent_result(updated)
595
596        # Protocol 2: runner.run(tenant_id, agent_type, task[, metadata])
597        if hasattr(self._runner, "run") and callable(self._runner.run):
598            params = inspect.signature(self._runner.run).parameters
599            if "metadata" in params:
600                result = self._runner.run(tenant_id, agent_type, task, metadata=metadata)
601            else:
602                result = self._runner.run(tenant_id, agent_type, task)
603            if asyncio.iscoroutine(result):
604                return await result
605            return result
606
607        # Protocol 3: runner is a coroutine function itself
608        if asyncio.iscoroutinefunction(self._runner):
609            kwargs = _runner_kwargs(self._runner, tenant_id, agent_type, task, metadata)
610            return await self._runner(**kwargs)
611
612        # Protocol 4: synchronous callable
613        if callable(self._runner):
614            kwargs = _runner_kwargs(self._runner, tenant_id, agent_type, task, metadata)
615            return self._runner(**kwargs)
616
617        raise TypeError(
618            f"agent_runner does not expose a callable interface: {type(self._runner)}"
619        )

132    norm2 = _normalize_sql(sql2)
133

138        import sqlparse  # type: ignore
139
140        parsed1 = sqlparse.parse(sql1)
141        parsed2 = sqlparse.parse(sql2)
142

144            return 0.0

159                for t in stmt.flatten()
