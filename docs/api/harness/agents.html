130    async def run(self, ctx: AgentContext) -> AgentResult:
131        """Execute the full agent lifecycle and return an AgentResult."""
132        run_start = time.monotonic()
133        output: str = ""
134        total_cost_usd: float = 0.0
135        tool_calls_total = 0
136        tool_errors_total = 0
137        guardrail_hits = 0
138        cache_hits = 0
139        cache_read_tokens = 0
140        metrics = _get_metrics()
141
142        # 1. Emit started event and begin MLflow run
143        await self._emit_event(StepEvent.started(ctx))
144        if metrics is not None:
145            try:
146                metrics.active_runs.labels(agent_type=self.agent_type).inc()
147            except Exception:
148                pass
149
150        async with self._mlflow_context(ctx):
151            try:
152                # Log run start locally
153                _append_run_log(ctx.run_id, {
154                    "event": "run_started",
155                    "agent_type": ctx.agent_type,
156                    "tenant_id": ctx.tenant_id,
157                    "task": ctx.task,
158                    "max_steps": ctx.max_steps,
159                    "max_tokens": ctx.max_tokens,
160                })
161
162                # 2. Resume from checkpoint if one exists
163                await self._maybe_resume_checkpoint(ctx)
164
165                # Main agentic loop
166                history: list[dict[str, Any]] = []
167
168                while ctx.is_budget_ok():
169                    # 3a. Fit history to context window
170                    history = await self._fit_history(ctx, history)
171
172                    # 3b. Build retrieval context from memory
173                    retrieval_context = await self._smart_retrieve(ctx)
174
175                    # 3c. Build messages
176                    messages = self.build_messages(ctx, history, retrieval_context)
177                    system_prompt = self.build_system_prompt(ctx)
178
179                    # 3d. LLM call with OTel span
180                    async with self._llm_span(ctx):
181                        response = await self._call_llm(ctx, messages, system_prompt)
182
183                    # 3e-f. Record token usage
184                    total_tokens = response.input_tokens + response.output_tokens
185                    ctx.tick(tokens=total_tokens)
186                    if response.cached:
187                        cache_hits += 1
188                        cache_read_tokens += response.input_tokens
189
190                    # 3g. Cost tracking
191                    if self._cost_tracker is not None:
192                        try:
193                            run_cost = await self._cost_tracker.record(
194                                run_id=ctx.run_id,
195                                tenant_id=ctx.tenant_id,
196                                model=response.model,
197                                input_tokens=response.input_tokens,
198                                output_tokens=response.output_tokens,
199                            )
200                            total_cost_usd += run_cost.cost_usd
201                        except Exception as exc:
202                            logger.warning("Cost tracking failed: %s", exc)
203
204                    # 3h. MLflow LLM span
205                    await self._log_llm_span(ctx, response)
206
207                    # 3i. Emit llm_called event
208                    await self._emit_event(StepEvent.llm_called(ctx, response))
209
210                    # Log LLM call locally
211                    _append_run_log(ctx.run_id, {
212                        "event": "llm_call",
213                        "step": ctx.step_count,
214                        "model": response.model,
215                        "provider": response.provider,
216                        "input_tokens": response.input_tokens,
217                        "output_tokens": response.output_tokens,
218                        "cached": response.cached,
219                        "has_tool_calls": bool(response.tool_calls),
220                        "response_preview": response.content[:300] if response.content else "",
221                    })
222
223                    # 3j. Safety check on output
224                    if self._safety_pipeline is not None:
225                        try:
226                            guard = await _safe_call(
227                                self._safety_pipeline.check_output,
228                                {"content": response.content},
229                            )
230                            if guard is not None and getattr(guard, "blocked", False):
231                                raise SafetyViolation(
232                                    f"Output blocked by safety pipeline: {getattr(guard, 'reason', '')}",
233                                    guard_source="output_guard",
234                                    failure_class=FailureClass.SAFETY_OUTPUT,
235                                )
236                        except SafetyViolation:
237                            raise
238                        except Exception as exc:
239                            logger.debug("Safety output check raised: %s", exc)
240
241                    # Add assistant message to history
242                    history.append({"role": "assistant", "content": response.content})
243
244                    # 3k. Push assistant response to short-term memory
245                    if ctx.memory is not None:
246                        try:
247                            await ctx.memory.push_message(
248                                run_id=ctx.run_id,
249                                role="assistant",
250                                content=response.content,
251                                tokens=response.output_tokens,
252                            )
253                        except Exception as exc:
254                            logger.debug("Failed to push assistant message to memory: %s", exc)
255
256                    # 3l. If no tool calls, we are done
257                    if not response.tool_calls:
258                        output = self.extract_final_answer(history)
259                        break
260                    tool_calls_total += len(response.tool_calls)
261
262                    # 3m. Execute tool calls — HITL sequential, execution parallel
263                    tool_results_for_history: list[dict[str, Any]] = []
264
265                    # HITL must be sequential (interactive human approval)
266                    for call in response.tool_calls:
267                        await self._check_hitl(ctx, call)
268
269                    # Execute all approved tool calls in parallel
270                    async def _execute_one(call: Any) -> ToolResult:
271                        try:
272                            return await self._tool_registry.execute(ctx, call)
273                        except ToolError as exc:
274                            logger.warning("Tool '%s' failed: %s", call.name, exc)
275                            return ToolResult(
276                                data=None,
277                                error=str(exc),
278                                metadata={"failure_class": exc.failure_class.value},
279                            )
280                        except SafetyViolation as exc:
281                            logger.warning("Tool '%s' blocked: %s", call.name, exc)
282                            return ToolResult(
283                                data=None,
284                                error=f"Blocked by safety policy: {exc}",
285                                metadata={"guardrail_hit": True},
286                            )
287
288                    tool_results: list[ToolResult] = list(
289                        await asyncio.gather(*[_execute_one(c) for c in response.tool_calls])
290                    )
291
292                    # Process results sequentially (history order must be preserved)
293                    for call, result in zip(response.tool_calls, tool_results, strict=True):
294                        # Push tool result to memory
295                        if result.is_error:
296                            tool_errors_total += 1
297                        if result.metadata.get("guardrail_hit") or (
298                            result.error
299                            and "blocked by safety policy" in result.error.lower()
300                        ):
301                            guardrail_hits += 1
302
303                        if ctx.memory is not None:
304                            try:
305                                await ctx.memory.push_message(
306                                    run_id=ctx.run_id,
307                                    role="tool",
308                                    content=result.to_text(),
309                                    tokens=0,
310                                )
311                            except Exception as mem_exc:
312                                logger.debug("Failed to push tool result to memory: %s", mem_exc)
313
314                        # Emit tool_called step event
315                        await self._emit_event(StepEvent.tool_called(ctx, call, result))
316
317                        # Log tool call locally
318                        _append_run_log(ctx.run_id, {
319                            "event": "tool_call",
320                            "step": ctx.step_count,
321                            "tool": call.name,
322                            "args": call.args,
323                            "success": not result.is_error,
324                            "error": result.error,
325                            "result_preview": str(result.data)[:300] if result.data else "",
326                        })
327
328                        # Audit log
329                        await self._audit(ctx, call, result)
330
331                        tool_results_for_history.append(
332                            {
333                                "role": "tool",
334                                "tool_use_id": call.id,
335                                "content": result.to_text(),
336                            }
337                        )
338
339                    # Append tool results to history
340                    history.extend(tool_results_for_history)
341
342                    # 3n. Checkpoint every N steps
343                    if ctx.step_count % _CHECKPOINT_INTERVAL == 0:
344                        await self._save_checkpoint(ctx, history)
345
346                else:
347                    # Budget exceeded — loop condition failed
348                    ctx.failed = True
349                    if ctx.step_count >= ctx.max_steps:
350                        ctx.failure_class = FailureClass.BUDGET_STEPS.value
351                    elif ctx.token_count >= ctx.max_tokens:
352                        ctx.failure_class = FailureClass.BUDGET_TOKENS.value
353                    else:
354                        ctx.failure_class = FailureClass.BUDGET_TIME.value
355                    await self._emit_event(
356                        StepEvent(
357                            run_id=ctx.run_id,
358                            step=ctx.step_count,
359                            event_type="budget_exceeded",
360                            payload={"failure_class": ctx.failure_class},
361                            timestamp=_utcnow(),
362                        )
363                    )
364
365            except BudgetExceeded as exc:
366                ctx.failed = True
367                ctx.failure_class = exc.failure_class.value
368                output = f"Budget exceeded: {exc}"
369                await self._record_failure(ctx, exc)
370                await self._emit_event(StepEvent.failed(ctx, str(exc)))
371
372            except HITLRejected as exc:
373                ctx.failed = True
374                ctx.failure_class = FailureClass.INTER_AGENT_REJECT.value
375                guardrail_hits += 1
376                output = f"HITL rejected: {exc}"
377                await self._record_failure(ctx, exc)
378                await self._emit_event(StepEvent.failed(ctx, str(exc)))
379
380            except SafetyViolation as exc:
381                ctx.failed = True
382                ctx.failure_class = exc.failure_class.value
383                guardrail_hits += 1
384                output = f"Safety violation: {exc}"
385                await self._record_failure(ctx, exc)
386                await self._emit_event(StepEvent.failed(ctx, str(exc)))
387
388            except HarnessError as exc:
389                ctx.failed = True
390                ctx.failure_class = exc.failure_class.value
391                output = f"Harness error: {exc}"
392                await self._record_failure(ctx, exc)
393                await self._emit_event(StepEvent.failed(ctx, str(exc)))
394
395            except asyncio.CancelledError:
396                ctx.failed = True
397                ctx.failure_class = FailureClass.UNKNOWN.value
398                output = "Run was cancelled."
399                await self._emit_event(StepEvent.failed(ctx, "cancelled"))
400                raise
401
402            except Exception as exc:
403                ctx.failed = True
404                ctx.failure_class = self._classify_exception(exc).value
405                output = f"Unexpected error: {exc}"
406                logger.exception("Unhandled exception in agent run %s", ctx.run_id)
407                await self._record_failure(ctx, exc)
408                await self._emit_event(StepEvent.failed(ctx, str(exc)))
409
410            finally:
411                # Save full conversation history to disk (gitignored, local only)
412                if history:
413                    _save_conversation(ctx.workspace_path, ctx.run_id, history)
414
415                # Feed online learning monitor so Hermes can detect regressions
416                if self._online_monitor is not None:
417                    try:
418                        version_id = ""
419                        version_number = 0
420                        if self._prompt_manager is not None:
421                            pv = await _safe_call(
422                                self._prompt_manager.get_version, self.agent_type
423                            )
424                            if pv is not None:
425                                version_id = pv.version_id
426                                version_number = pv.version_number
427                        if version_id:
428                            await self._online_monitor.record_run(
429                                agent_type=self.agent_type,
430                                version_id=version_id,
431                                version_number=version_number,
432                                success=not ctx.failed,
433                                cost_usd=total_cost_usd,
434                                steps=ctx.step_count,
435                            )
436                    except Exception as exc:
437                        logger.debug("online_monitor.record_run failed: %s", exc)
438
439                # Log run completion locally
440                elapsed = time.monotonic() - run_start
441                _append_run_log(ctx.run_id, {
442                    "event": "run_finished",
443                    "agent_type": ctx.agent_type,
444                    "success": not ctx.failed,
445                    "failure_class": ctx.failure_class,
446                    "total_steps": ctx.step_count,
447                    "total_tokens": ctx.token_count,
448                    "total_cost_usd": total_cost_usd,
449                    "elapsed_seconds": round(elapsed, 2),
450                    "output_preview": output[:300] if output else "",
451                })
452
453                # 8. Decrement active_runs gauge
454                if metrics is not None:
455                    try:
456                        metrics.active_runs.labels(agent_type=self.agent_type).dec()
457                        metrics.agent_runs_total.labels(
458                            agent_type=self.agent_type,
459                            success=str(not ctx.failed).lower(),
460                        ).inc()
461                    except Exception:
462                        pass
463
464        # Emit completed event if success
465        if not ctx.failed:
466            await self._emit_event(StepEvent.completed(ctx, output))
467
468        elapsed = time.monotonic() - run_start
469
470        return AgentResult(
471            run_id=ctx.run_id,
472            output=output,
473            steps=ctx.step_count,
474            tokens=ctx.token_count,
475            success=not ctx.failed,
476            failure_class=ctx.failure_class,
477            error_message=output if ctx.failed else None,
478            elapsed_seconds=elapsed,
479            cost_usd=total_cost_usd,
480            tool_calls=tool_calls_total,
481            tool_errors=tool_errors_total,
482            guardrail_hits=guardrail_hits,
483            handoff_count=int(ctx.metadata.get("handoff_count", 0) or 0),
484            cache_hits=cache_hits,
485            cache_read_tokens=cache_read_tokens,
486        )

492    def build_system_prompt(self, ctx: AgentContext) -> str:
493        """Return the system prompt for this agent. Override in subclasses."""
494        return (
495            f"You are a helpful AI agent of type '{self.agent_type}'. "
496            f"Task: {ctx.task}\n"
497            "Use the available tools to complete the task. "
498            "When you have a final answer, respond without requesting any more tools."

501    def build_messages(
502        self,
503        ctx: AgentContext,
504        history: list[dict[str, Any]],
505        retrieval_context: str,
506    ) -> list[dict[str, Any]]:
507        """Assemble the messages list for the LLM call."""
508        messages: list[dict[str, Any]] = []
509
510        # If we have retrieval context, prepend it as a user message
511        if retrieval_context and not history:
512            messages.append(
513                {
514                    "role": "user",
515                    "content": (
516                        f"Relevant context from memory:\n{retrieval_context}\n\n"
517                        f"Task: {ctx.task}"
518                    ),
519                }
520            )
521        elif not history:
522            messages.append({"role": "user", "content": ctx.task})
523
524        messages.extend(history)
525
526        # Ensure first message is from user
527        if messages and messages[0]["role"] != "user":
528            messages.insert(0, {"role": "user", "content": ctx.task})
529
530        return messages

532    def extract_final_answer(self, history: list[dict[str, Any]]) -> str:
533        """Extract the final answer text from agent history."""
534        # Walk history in reverse to find the last assistant message
535        for msg in reversed(history):
536            if msg.get("role") == "assistant":
537                content = msg.get("content", "")
538                if isinstance(content, list):
539                    # Anthropic block content
540                    text_parts = [
541                        b.get("text", "") if isinstance(b, dict) else str(b)
542                        for b in content
543                        if not (isinstance(b, dict) and b.get("type") == "tool_use")
544                    ]
545                    text = " ".join(text_parts).strip()
546                    if text:
547                        return text
548                elif isinstance(content, str) and content.strip():
549                    return content.strip()
550        return "Task completed."

552    def _classify_exception(self, e: Exception) -> FailureClass:
553        """Map an exception to a FailureClass."""
554        if isinstance(e, BudgetExceeded):
555            return e.failure_class
556        if isinstance(e, ToolError):
557            return e.failure_class
558        if isinstance(e, SafetyViolation):
559            return e.failure_class
560        if isinstance(e, HITLRejected):
561            return FailureClass.INTER_AGENT_REJECT
562        if isinstance(e, HarnessError):
563            return e.failure_class
564        if isinstance(e, TimeoutError):
565            return FailureClass.BUDGET_TIME
566        return FailureClass.UNKNOWN

572    async def _check_hitl(self, ctx: AgentContext, call: ToolCall) -> None:
573        """Check if human-in-the-loop approval is required for this tool call."""
574        # Policy-based check: look for hitl manager in metadata
575        hitl_manager = ctx.metadata.get("hitl_manager")
576        policy = ctx.metadata.get("policy")
577
578        if hitl_manager is None or policy is None:
579            return
580
581        if not policy.requires_hitl(call.name):
582            return
583
584        logger.info(
585            "HITL required for tool '%s' in run '%s'", call.name, ctx.run_id
586        )
587
588        # Create approval request
589        request = await hitl_manager.request_approval(
590            ctx=ctx,
591            tool_name=call.name,
592            tool_args=call.args,
593        )
594
595        # Await the decision
596        decision = await hitl_manager.await_decision(
597            request_id=request.request_id,
598            timeout=3600.0,
599        )
600
601        if decision == "rejected":
602            raise HITLRejected(
603                f"Human reviewer rejected tool call '{call.name}'",
604                request_id=request.request_id,
605            )
606        if decision == "expired":
607            raise HITLRejected(
608                f"HITL approval for '{call.name}' expired",
609                request_id=request.request_id,
610            )

616    async def _call_llm(
617        self,
618        ctx: AgentContext,
619        messages: list[dict[str, Any]],
620        system: str,
621    ) -> LLMResponse:
622        """Call the LLM router with tools from the registry."""
623        tools = (
624            self._tool_registry.to_anthropic_format()
625            if self._tool_registry is not None
626            else []
627        )
628
629        remaining_tokens = ctx.max_tokens - ctx.token_count
630        max_tokens = min(4096, max(256, remaining_tokens // 2))
631
632        # Use streaming when requested and no tools are needed (streaming + tools
633        # requires special handling that varies by provider)
634        if ctx.metadata.get("stream_tokens") and not tools:
635            return await self._call_llm_streaming(ctx, messages, system, max_tokens)
636
637        return await self._llm_router.complete(
638            messages=messages,
639            system=system,
640            tools=tools if tools else None,
641            max_tokens=max_tokens,
642        )
643
644    async def _call_llm_streaming(
645        self,
646        ctx: AgentContext,
647        messages: list[dict[str, Any]],
648        system: str,
649        max_tokens: int,
650    ) -> LLMResponse:
651        """Stream tokens from the LLM, publishing each delta as a StepEvent.
652
653        Accumulates the full text and returns a synthetic LLMResponse so the
654        rest of the agent loop (history, cost tracking, safety) works unchanged.
655        """
656        chunks: list[str] = []
657        try:
658            async for delta in self._llm_router.stream(
659                messages=messages,
660                system=system,
661                max_tokens=max_tokens,
662            ):
663                chunks.append(delta)
664                # Publish token delta for SSE clients
665                await self._emit_event(
666                    StepEvent(
667                        run_id=ctx.run_id,
668                        step=ctx.step_count,
669                        event_type="token_delta",
670                        payload={"delta": delta},
671                        timestamp=_utcnow(),
672                    )
673                )
674        except Exception as exc:
675            logger.warning("Streaming LLM call failed, falling back to complete(): %s", exc)
676            return await self._llm_router.complete(
677                messages=messages,
678                system=system,
679                max_tokens=max_tokens,
680            )
681
682        full_text = "".join(chunks)
683        # Estimate tokens (streaming responses don't always return exact counts)
684        estimated_tokens = max(1, len(full_text) // 4)
685        return LLMResponse(
686            content=full_text,
687            tool_calls=[],
688            input_tokens=estimated_tokens,
689            output_tokens=estimated_tokens,
690            model=getattr(self._llm_router, "model", "unknown"),
691            provider=getattr(self._llm_router, "provider_name", "unknown"),
692            cached=False,
693        )

699    async def _smart_retrieve(self, ctx: AgentContext) -> str:
700        """Retrieve relevant context from long-term memory for the current task."""
701        if ctx.memory is None:
702            return ""
703        try:
704            if hasattr(ctx.memory, "smart_retrieve"):
705                result = await ctx.memory.smart_retrieve(
706                    query=ctx.task,
707                    run_id=ctx.run_id,
708                    k=5,
709                )
710                if result:
711                    if isinstance(result, str):
712                        return result
713                    if hasattr(result, "formatted"):
714                        return result.formatted
715                    return str(result)
716        except Exception as exc:
717            logger.debug("smart_retrieve failed: %s", exc)
718        return ""

720    async def _fit_history(
721        self,
722        ctx: AgentContext,
723        history: list[dict[str, Any]],
724    ) -> list[dict[str, Any]]:
725        """Trim history to fit within the context window budget.
726
727        When history exceeds _MAX_HISTORY_MESSAGES, the older portion is
728        summarized via LLM call and replaced with a single summary message.
729        Keeps the last _RECENT_MESSAGES_KEEP messages verbatim.
730        Falls back to a naive slice if summarization fails.
731        """
732        if len(history) <= _MAX_HISTORY_MESSAGES:
733            return history
734
735        recent = history[-_RECENT_MESSAGES_KEEP:]
736        older = history[:-_RECENT_MESSAGES_KEEP]
737
738        # Try memory manager's fit_history first (most efficient path)
739        if ctx.memory is not None and hasattr(ctx.memory, "fit_history"):
740            try:
741                fit_result = await ctx.memory.fit_history(
742                    run_id=ctx.run_id,
743                    history=older,
744                    keep_last=0,
745                )
746                summary_text = getattr(fit_result, "summary", None)
747                if summary_text:
748                    logger.info(

 51    async def _populate_schema(self, ctx: AgentContext) -> None:
 52        """Introspect the database and store schema in graph memory.
 53
 54        This is a one-time operation per session, gated by ctx.metadata["schema_loaded"].
 55        Subsequent calls within the same session are no-ops.
 56        """
 57        if ctx.metadata.get("schema_loaded"):
 58            return
 59
 60        # Mark as loading to prevent concurrent calls
 61        ctx.metadata["schema_loaded"] = True
 62
 63        logger.info(
 64            "Populating database schema into graph memory for run %s", ctx.run_id
 65        )
 66
 67        # Try to get list_tables and describe_table from tool registry
 68        if self._tool_registry is None:
 69            logger.debug("No tool_registry — skipping schema population")
 70            return
 71
 72        list_tables_tool = self._tool_registry.get("list_tables")
 73        describe_table_tool = self._tool_registry.get("describe_table")
 74
 75        if list_tables_tool is None:
 76            logger.debug("list_tables tool not registered — skipping schema population")
 77            return
 78
 79        # Create a lightweight ToolCall to invoke list_tables
 80        from harness.core.context import ToolCall
 81        import uuid
 82
 83        list_call = ToolCall(
 84            id=uuid.uuid4().hex,
 85            name="list_tables",
 86            args={},
 87        )
 88
 89        try:
 90            list_result = await list_tables_tool.execute(ctx, list_call.args)
 91        except Exception as exc:
 92            logger.warning("list_tables during schema population failed: %s", exc)
 93            return
 94
 95        if list_result.is_error or not list_result.data:
 96            logger.debug("list_tables returned no data: %s", list_result.error)
 97            return
 98
 99        tables: list[dict] = list_result.data if isinstance(list_result.data, list) else []
100        table_names = [
101            t.get("table_name") or t.get("name")
102            for t in tables
103            if t.get("table_name") or t.get("name")
104        ]
105
106        logger.info("Discovered %d tables for schema population", len(table_names))
107
108        # Populate graph memory with table and column nodes
109        graph_rag_available = False
110        if ctx.memory is not None and hasattr(ctx.memory, "graph_rag"):
111            graph_rag_available = True
112
113        for table_name in table_names[:50]:  # cap at 50 tables
114            # Add table node to graph memory
115            if ctx.memory is not None and hasattr(ctx.memory, "graph") and ctx.memory.graph is not None:
116                try:
117                    await ctx.memory.graph.add_node(
118                        id=f"table:{table_name}",
119                        type="Table",
120                        props={"name": table_name, "run_id": ctx.run_id},
121                    )
122                except Exception as exc:
123                    logger.debug("Failed to add table node '%s': %s", table_name, exc)
124
125            # Describe table to get columns
126            if describe_table_tool is None:
127                continue
128
129            describe_call = ToolCall(
130                id=uuid.uuid4().hex,
131                name="describe_table",
132                args={"table_name": table_name},
133            )
134
135            try:
136                desc_result = await describe_table_tool.execute(ctx, describe_call.args)
137                if desc_result.is_error or not desc_result.data:
138                    continue
139
140                table_info = desc_result.data
141                columns = table_info.get("columns", []) if isinstance(table_info, dict) else []
142
143                for col in columns:
144                    col_name = col.get("column_name") or col.get("name")
145                    if not col_name:
146                        continue
147                    if ctx.memory is not None and hasattr(ctx.memory, "graph") and ctx.memory.graph is not None:
148                        try:
149                            await ctx.memory.graph.add_node(
150                                id=f"column:{table_name}.{col_name}",
151                                type="Column",
152                                props={
153                                    "name": col_name,
154                                    "table": table_name,
155                                    "data_type": col.get("data_type", "unknown"),
156                                    "nullable": col.get("is_nullable", "YES"),
157                                },
158                            )
159                            await ctx.memory.graph.add_edge(
160                                src=f"table:{table_name}",
161                                tgt=f"column:{table_name}.{col_name}",
162                                type="HAS_COLUMN",
163                            )
164                        except Exception as exc:
165                            logger.debug(
166                                "Failed to add column node '%s.%s': %s",
167                                table_name,
168                                col_name,
169                                exc,
170                            )
171
172                # Also store schema description in vector memory for RAG
173                if ctx.memory is not None and hasattr(ctx.memory, "push_to_vector"):
174                    col_summary = ", ".join(
175                        f"{c.get('column_name', c.get('name', '?'))}({c.get('data_type', '?')})"
176                        for c in columns[:20]
177                    )
178                    schema_text = f"Table: {table_name}\nColumns: {col_summary}"
179                    try:
180                        await ctx.memory.push_to_vector(
181                            run_id=ctx.run_id,
182                            text=schema_text,
183                            metadata={"type": "schema", "table": table_name},
184                        )
185                    except Exception as exc:
186                        logger.debug("Failed to push schema to vector: %s", exc)
187
188            except Exception as exc:
189                logger.warning(
190                    "describe_table failed for '%s' during schema population: %s",
191                    table_name,
192                    exc,
193                )
194                continue
195
196        # Store schema summary in ctx metadata
197        ctx.metadata["schema_tables"] = table_names
198        logger.info(
199            "Schema population complete: %d tables stored in graph memory",
200            len(table_names),
201        )
