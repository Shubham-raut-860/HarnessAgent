 57class Evaluator:
 58    """Evaluates a patch by replaying failing tasks with the patch applied.
 59
 60    For each test case (an ErrorRecord), the evaluator:
 61    1. Builds a patched config by applying the patch to the base config.
 62    2. Creates and runs an agent on the original task with the patched config.
 63    3. Compares success/failure and resource usage against the original.
 64
 65    The composite score rewards success, brevity (fewer steps), and
 66    token efficiency (fewer tokens).
 67    """
 68
 69    def __init__(self, agent_runner: Any, error_collector: Any) -> None:
 70        """
 71        Args:
 72            agent_runner:    AgentRunner instance used to create/execute test runs.
 73            error_collector: ErrorCollector for sampling and resolving records.
 74        """
 75        self._runner = agent_runner
 76        self._error_collector = error_collector
 77
 78    async def score(
 79        self,
 80        patch: Any,  # Patch
 81        test_cases: list[Any],  # list[ErrorRecord]
 82        agent_type: str,
 83    ) -> EvalResult:
 84        """Evaluate a patch across a set of test cases.
 85
 86        Args:
 87            patch:      The Patch to evaluate.
 88            test_cases: ErrorRecord objects whose tasks will be replayed.
 89            agent_type: The agent type for these test cases.
 90
 91        Returns:
 92            An EvalResult with composite score.
 93        """
 94        if not test_cases:
 95            logger.info("No test cases for patch %s — returning zero score", getattr(patch, "patch_id", "?"))
 96            return EvalResult(
 97                patch_id=getattr(patch, "patch_id", ""),
 98                test_cases=0,
 99                successes=0,
100                failures=0,
101            )
102
103        # Get the base config
104        base_config = await self._get_base_config(agent_type)
105        patched_config = await self._apply_patch_to_config(base_config, patch)
106
107        successes = 0
108        failures = 0
109        step_deltas: list[float] = []
110        token_deltas: list[float] = []
111
112        tasks = [
113            self._run_test_case(tc, patched_config, agent_type)
114            for tc in test_cases
115        ]
116        results = await asyncio.gather(*tasks, return_exceptions=True)
117
118        for i, result in enumerate(results):
119            tc = test_cases[i]
120            if isinstance(result, Exception):
121                logger.warning("Test case %d failed with exception: %s", i, result)
122                failures += 1
123                continue
124
125            run_result, baseline_steps, baseline_tokens = result
126
127            if run_result is None or not getattr(run_result, "success", False):
128                failures += 1
129            else:
130                successes += 1
131                # Compare resource usage to baseline
132                new_steps = getattr(run_result, "steps", 0)
133                new_tokens = getattr(run_result, "tokens", 0)
134                orig_steps = getattr(tc, "context_snapshot", {}).get("step_count", baseline_steps)
135                orig_tokens = getattr(tc, "context_snapshot", {}).get("token_count", baseline_tokens)
136
137                if orig_steps > 0:
138                    step_deltas.append(new_steps - orig_steps)
139                if orig_tokens > 0:
140                    token_deltas.append(new_tokens - orig_tokens)
141
142        total = len(test_cases)
143        avg_steps_delta = sum(step_deltas) / len(step_deltas) if step_deltas else 0.0
144        avg_tokens_delta = sum(token_deltas) / len(token_deltas) if token_deltas else 0.0
145        success_rate = successes / total if total > 0 else 0.0
146
147        # Composite score: higher is better
148        # Penalty for more steps and more tokens
149        raw_score = success_rate - 0.01 * avg_steps_delta - 0.001 * avg_tokens_delta
150        score = max(0.0, min(1.0, raw_score))
151
152        patch_id = getattr(patch, "patch_id", "")
153        logger.info(
154            "Patch %s scored %.3f: %d/%d success, steps_delta=%.1f, tokens_delta=%.1f",
155            patch_id[:8],
156            score,
157            successes,
158            total,
159            avg_steps_delta,
160            avg_tokens_delta,
161        )
162
163        return EvalResult(
164            patch_id=patch_id,
165            test_cases=total,
166            successes=successes,
167            failures=failures,
168            avg_steps_delta=avg_steps_delta,
169            avg_tokens_delta=avg_tokens_delta,
170            score=score,
171        )

325        delta = patched_sr - baseline_sr

338        if patched_sr < baseline_sr:

340                await self._pm.rollback(agent_type, steps=1)
341                logger.info("Rolled back patch (regression detected)")
342            except Exception as exc:
343                logger.warning("Rollback after regression failed: %s", exc)
