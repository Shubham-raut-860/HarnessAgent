 75
 76    Processes chunks in batches to reduce LLM calls.  Each batch combines
 77    multiple chunks with clear separators.  Parses the response as a JSON
 78    array of triple dicts.
 79
 80    Args:
 81        chunks:       List of Chunk objects to extract from.
 82        llm_provider: An LLMProvider-compatible object.
 83        batch_size:   Number of chunks to combine per LLM call.
 84
 85    Returns:
 86        List of ExtractedFact objects across all chunks.
 87    """
 88    all_facts: list[ExtractedFact] = []
 89
 90    # Process in batches
 91    for batch_start in range(0, len(chunks), batch_size):
 92        batch = chunks[batch_start: batch_start + batch_size]
 93        combined_text = "\n\n---\n\n".join(
 94            f"[Chunk {i + 1}/{len(batch)}]\n{c.content}" for i, c in enumerate(batch)
 95        )
 96        # Truncate to avoid context overflows (approx 8000 chars ≈ 2000 tokens)
 97        if len(combined_text) > 8000:
 98            combined_text = combined_text[:8000] + "\n...[truncated]"
 99
100        prompt = _EXTRACT_PROMPT_TEMPLATE.format(text=combined_text)
101        try:
102            response = await llm_provider.complete(
103                messages=[{"role": "user", "content": prompt}],
104                max_tokens=1024,
105                system=_EXTRACT_SYSTEM,
106            )
107            raw = response.content.strip()
108
109            # Strip markdown code fences
110            if raw.startswith("```"):
111                raw = re.sub(r"^```[a-z]*\n?", "", raw, flags=re.MULTILINE)
112                raw = re.sub(r"\n?```$", "", raw, flags=re.MULTILINE)
113                raw = raw.strip()
114
115            parsed = json.loads(raw)
116            if not isinstance(parsed, list):
117                logger.warning("LLM extraction returned non-list; skipping batch")
118                continue
119
120            # Map chunk IDs to batch chunks for attribution
121            for item in parsed:
122                if not isinstance(item, dict):
123                    continue
124                subject = str(item.get("subject", "")).strip()
125                predicate = str(item.get("predicate", "")).strip()
126                obj = str(item.get("object", "")).strip()
127                if not subject or not predicate or not obj:
128                    continue
129                confidence = float(item.get("confidence", 1.0))
130                confidence = max(0.0, min(1.0, confidence))
131
132                # Attribute to first chunk in batch as a representative
133                source_chunk_id = batch[0].chunk_id if batch else ""
134
135                all_facts.append(
136                    ExtractedFact(
137                        subject=subject,
138                        predicate=predicate,
139                        object_=obj,
140                        confidence=confidence,
141                        source_chunk_id=source_chunk_id,
142                    )
143                )

145        except json.JSONDecodeError as exc:

150            )

154                batch_start,
155                exc,
156            )

def extract_sql_schema_facts( tables_info: list[dict]) -> list[ExtractedFact]: 
167def extract_sql_schema_facts(tables_info: list[dict]) -> list[ExtractedFact]:
168    """Extract structural knowledge triples from SQL schema metadata.
169
170    Creates facts like:
171    - (table_name, "has_column", column_name)
172    - (table_name, "column_type", "col_name:type")
173    - (table_name, "joins", other_table) when FK relationships are detected
174    - (table_name, "has_primary_key", pk_column)
175
176    Args:
177        tables_info: List of dicts, each with keys:
178            - "table_name" (str)
179            - "columns" (list[dict] with "name", "type", optional "primary_key", "foreign_key")
180            - Optional "foreign_keys" (list[dict] with "column", "references_table", "references_column")
181
182    Returns:
183        List of ExtractedFact objects.
184    """
185    facts: list[ExtractedFact] = []
186
187    for table_info in tables_info:
188        table_name = table_info.get("table_name", "")
189        if not table_name:
190            continue
191
192        columns = table_info.get("columns", [])
193        for col in columns:
194            col_name = col.get("name", "")
195            col_type = col.get("type", "")
196            if not col_name:
197                continue
198
199            facts.append(
200                ExtractedFact(
201                    subject=table_name,
202                    predicate="has_column",
203                    object_=col_name,
204                    confidence=1.0,
205                )
206            )
207
208            if col_type:
209                facts.append(
210                    ExtractedFact(
211                        subject=table_name,
212                        predicate="column_type",
213                        object_=f"{col_name}:{col_type}",
214                        confidence=1.0,
215                    )
216                )
217
218            # Primary key detection
219            if col.get("primary_key") or col.get("is_primary_key"):
220                facts.append(
221                    ExtractedFact(
222                        subject=table_name,
223                        predicate="has_primary_key",
224                        object_=col_name,
225                        confidence=1.0,
226                    )
227                )
228
229        # Foreign key / join detection
230        foreign_keys = table_info.get("foreign_keys", [])
231        for fk in foreign_keys:
232            fk_col = fk.get("column", "")
233            ref_table = fk.get("references_table", "")
234            ref_col = fk.get("references_column", "")
235            if fk_col and ref_table:
236                facts.append(
237                    ExtractedFact(
238                        subject=table_name,
239                        predicate="joins",
240                        object_=ref_table,
241                        confidence=1.0,
242                    )
243                )
244                if ref_col:
245                    facts.append(
246                        ExtractedFact(
247                            subject=table_name,
248                            predicate="foreign_key",
249                            object_=f"{fk_col} -> {ref_table}.{ref_col}",
250                            confidence=1.0,
251                        )
252                    )
253
254        # Also check inline column foreign key hints
255        for col in columns:
256            col_fk = col.get("foreign_key", "")
257            if col_fk and "." in col_fk:
258                ref_parts = col_fk.split(".", 1)
259                ref_table = ref_parts[0]
260                facts.append(
261                    ExtractedFact(
262                        subject=table_name,
263                        predicate="joins",
264                        object_=ref_table,
265                        confidence=0.9,
266                    )
267                )
268
269    logger.debug(
270        "Extracted %d schema facts from %d tables", len(facts), len(tables_info)
271    )
272    return facts

 85    async def ingest(
 86        self,
 87        source: str,
 88        tenant_id: str,
 89        agent_type: str = "research",
 90        extract_facts: bool = True,
 91        chunk_size: int = 500,
 92        chunk_overlap: int = 50,
 93    ) -> IngestionResult:
 94        """Ingest a single file or URL.
 95
 96        Pipeline:
 97            1. load(source)  → list[Document]
 98            2. chunk_documents(docs) → list[Chunk]
 99            3. For each chunk: memory_manager.remember(chunk.content, metadata, tenant_id)
100            4. If extract_facts and llm_provider: extract triples and store via add_fact
101
102        Args:
103            source:         File path or URL to ingest.
104            tenant_id:      Tenant to associate all memories with.
105            agent_type:     Tag used for routing/filtering in the memory store.
106            extract_facts:  Whether to run LLM-based entity extraction.
107            chunk_size:     Target chunk size in tokens.
108            chunk_overlap:  Overlap between consecutive chunks in tokens.
109
110        Returns:
111            IngestionResult summarising what was done.
112        """
113        result = IngestionResult(source=source)
114        t_start = time.monotonic()
115
116        # Step 1: Load documents
117        try:
118            docs = await load(source)
119            result.documents_loaded = len(docs)
120            if not docs:
121                result.errors.append(f"No documents loaded from: {source}")
122                result.duration_seconds = time.monotonic() - t_start
123                return result
124        except Exception as exc:
125            msg = f"Failed to load {source}: {exc}"
126            logger.error(msg)
127            result.errors.append(msg)
128            result.duration_seconds = time.monotonic() - t_start
129            return result
130
131        logger.info(
132            "Loaded %d documents from %s (agent_type=%s, tenant=%s)",
133            len(docs), source, agent_type, tenant_id,
134        )
135
136        # Step 2: Chunk documents
137        try:
138            chunks = chunk_documents(
139                docs,
140                chunk_size=chunk_size,
141                overlap=chunk_overlap,
142            )
143            result.chunks_created = len(chunks)
144        except Exception as exc:
145            msg = f"Chunking failed for {source}: {exc}"
146            logger.error(msg)
147            result.errors.append(msg)
148            result.duration_seconds = time.monotonic() - t_start
149            return result
150
151        logger.info("Created %d chunks from %s", len(chunks), source)
152
153        # Step 3: Store embeddings
154        embedded_count = 0
155        for chunk in chunks:
156            chunk_meta = {
157                **chunk.metadata,
158                "chunk_id": chunk.chunk_id,
159                "doc_id": chunk.doc_id,
160                "agent_type": agent_type,
161                "source": source,
162                "start_char": chunk.start_char,
163                "end_char": chunk.end_char,
164            }
165            try:
166                await self._remember(chunk.content, chunk_meta, tenant_id)
167                embedded_count += 1
168            except Exception as exc:
169                msg = f"Failed to store chunk {chunk.chunk_id}: {exc}"
170                logger.warning(msg)
171                result.errors.append(msg)
172
173        result.embeddings_stored = embedded_count
174        logger.info("Stored %d embeddings for %s", embedded_count, source)
175
176        # Step 4: Extract and store facts
177        if extract_facts and self._llm is not None:
178            try:
179                facts = await _extract_facts(
180                    chunks=chunks,
181                    llm_provider=self._llm,
182                )
183                facts_stored = 0
184                for fact in facts:
185                    try:
186                        await self._add_fact(
187                            subject=fact.subject,
188                            predicate=fact.predicate,
189                            object_=fact.object_,
190                            tenant_id=tenant_id,
191                        )
192                        facts_stored += 1
193                    except Exception as exc:
194                        logger.warning("Failed to store fact %s: %s", fact, exc)
195                result.facts_extracted = facts_stored
196                logger.info("Stored %d facts from %s", facts_stored, source)
197            except Exception as exc:
198                msg = f"Fact extraction failed for {source}: {exc}"
199                logger.warning(msg)
200                result.errors.append(msg)
201
202        result.duration_seconds = time.monotonic() - t_start
203        return result

209    async def ingest_directory(
210        self,
211        directory: str,
212        tenant_id: str,
213        concurrency: int = 3,
214        **kwargs,
215    ) -> list[IngestionResult]:
216        """Ingest all supported files in a directory concurrently.
217
218        Args:
219            directory:   Path to the directory to scan.
220            tenant_id:   Tenant to associate memories with.
221            concurrency: Max concurrent ingestion tasks.
222            **kwargs:    Forwarded to self.ingest() per file.
223
224        Returns:
225            List of IngestionResult, one per discovered file.
226        """
227        dir_path = Path(directory)
228        if not dir_path.is_dir():
229            logger.error("Directory not found: %s", directory)
230            return [
231                IngestionResult(
232                    source=directory,
233                    errors=[f"Directory not found: {directory}"],
234                )
235            ]
236
237        # Find all supported files recursively
238        files: list[Path] = []
239        for p in dir_path.rglob("*"):
240            if p.is_file() and p.suffix.lower() in _SUPPORTED_EXTENSIONS:
241                files.append(p)
242
243        if not files:
244            logger.warning("No supported files found in directory: %s", directory)
245            return []
246
247        logger.info(
248            "Ingesting %d files from directory %s (concurrency=%d)",
249            len(files), directory, concurrency,
250        )
251
252        semaphore = asyncio.Semaphore(concurrency)
253        tasks = [
254            self._ingest_with_semaphore(str(f), tenant_id, semaphore, **kwargs)
255            for f in files
256        ]
257        results = await asyncio.gather(*tasks, return_exceptions=True)
258

 85    async def ingest(
 86        self,
 87        source: str,
 88        tenant_id: str,
 89        agent_type: str = "research",
 90        extract_facts: bool = True,
 91        chunk_size: int = 500,
 92        chunk_overlap: int = 50,
 93    ) -> IngestionResult:
 94        """Ingest a single file or URL.
 95
 96        Pipeline:
 97            1. load(source)  → list[Document]
 98            2. chunk_documents(docs) → list[Chunk]
 99            3. For each chunk: memory_manager.remember(chunk.content, metadata, tenant_id)
100            4. If extract_facts and llm_provider: extract triples and store via add_fact
101
102        Args:
103            source:         File path or URL to ingest.
104            tenant_id:      Tenant to associate all memories with.
105            agent_type:     Tag used for routing/filtering in the memory store.
106            extract_facts:  Whether to run LLM-based entity extraction.
107            chunk_size:     Target chunk size in tokens.
108            chunk_overlap:  Overlap between consecutive chunks in tokens.
109
110        Returns:
111            IngestionResult summarising what was done.
112        """
113        result = IngestionResult(source=source)
114        t_start = time.monotonic()
115
116        # Step 1: Load documents
117        try:
118            docs = await load(source)
119            result.documents_loaded = len(docs)
120            if not docs:
121                result.errors.append(f"No documents loaded from: {source}")
122                result.duration_seconds = time.monotonic() - t_start
123                return result
124        except Exception as exc:
125            msg = f"Failed to load {source}: {exc}"
126            logger.error(msg)
127            result.errors.append(msg)
128            result.duration_seconds = time.monotonic() - t_start
129            return result
130
131        logger.info(
132            "Loaded %d documents from %s (agent_type=%s, tenant=%s)",
133            len(docs), source, agent_type, tenant_id,
134        )
135
136        # Step 2: Chunk documents
137        try:
138            chunks = chunk_documents(
139                docs,
140                chunk_size=chunk_size,
141                overlap=chunk_overlap,
142            )
143            result.chunks_created = len(chunks)
144        except Exception as exc:
145            msg = f"Chunking failed for {source}: {exc}"
146            logger.error(msg)
147            result.errors.append(msg)
148            result.duration_seconds = time.monotonic() - t_start
149            return result
150
151        logger.info("Created %d chunks from %s", len(chunks), source)
152
153        # Step 3: Store embeddings
154        embedded_count = 0
155        for chunk in chunks:
156            chunk_meta = {
157                **chunk.metadata,
158                "chunk_id": chunk.chunk_id,
159                "doc_id": chunk.doc_id,
160                "agent_type": agent_type,
161                "source": source,
162                "start_char": chunk.start_char,
163                "end_char": chunk.end_char,
164            }
165            try:
166                await self._remember(chunk.content, chunk_meta, tenant_id)
167                embedded_count += 1
168            except Exception as exc:
169                msg = f"Failed to store chunk {chunk.chunk_id}: {exc}"
170                logger.warning(msg)
171                result.errors.append(msg)
172
173        result.embeddings_stored = embedded_count
174        logger.info("Stored %d embeddings for %s", embedded_count, source)
175
176        # Step 4: Extract and store facts
177        if extract_facts and self._llm is not None:
178            try:
179                facts = await _extract_facts(
180                    chunks=chunks,
181                    llm_provider=self._llm,
182                )
183                facts_stored = 0
184                for fact in facts:
185                    try:
186                        await self._add_fact(
187                            subject=fact.subject,
188                            predicate=fact.predicate,
189                            object_=fact.object_,
190                            tenant_id=tenant_id,
191                        )
192                        facts_stored += 1
193                    except Exception as exc:
194                        logger.warning("Failed to store fact %s: %s", fact, exc)
195                result.facts_extracted = facts_stored
196                logger.info("Stored %d facts from %s", facts_stored, source)
197            except Exception as exc:
198                msg = f"Fact extraction failed for {source}: {exc}"
199                logger.warning(msg)
200                result.errors.append(msg)
201
202        result.duration_seconds = time.monotonic() - t_start
203        return result
