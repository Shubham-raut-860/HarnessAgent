harness.adapters

Framework adapters for HarnessAgent.

Each adapter wraps an external multi-agent framework so it can execute inside the harness lifecycle and receive observability, memory, safety, and cost tracking for free.

Supported frameworks

  • LangGraph — LangGraphAdapter wraps a StateGraph or compiled graph and streams astream() output as StepEvent objects.
  • AutoGen — AutoGenAdapter wraps a ConversableAgent pair (or GroupChat) and emits one event per conversation message.
  • CrewAI — CrewAIAdapter wraps a Crew and emits one event per agent step captured via step_callback.

Typical usage::

from harness.adapters import LangGraphAdapter, FrameworkResult

adapter = LangGraphAdapter(graph=my_compiled_graph)
async for event in adapter.run(ctx, input={"messages": [...]}):
    print(event.event_type, event.payload)

result: FrameworkResult = await adapter.get_result()
print(result.output)

Framework packages (langgraph, pyautogen / autogen, crewai) are optional runtime dependencies. An ImportError with a helpful install message is raised only when the adapter is actually used.

 64    async def run(
 65        self,
 66        ctx: "AgentContext",
 67        input: dict,  # noqa: A002
 68    ) -> AsyncIterator["StepEvent"]:
 69        """Stream the graph and yield one StepEvent per node execution.
 70
 71        Raises:
 72            ImportError: If ``langgraph`` is not installed.
 73            BudgetExceeded: Propagated from ``ctx.tick()`` when a budget limit
 74                is crossed (step count, token count, or elapsed time).
 75        """
 76        try:
 77            from langgraph.graph import StateGraph  # noqa: F401 — import check only
 78        except ImportError as exc:
 79            raise ImportError(
 80                "LangGraph is not installed.  Install it with:  pip install langgraph"
 81            ) from exc
 82
 83        # Compile lazily if a raw StateGraph was supplied.
 84        compiled = (
 85            self._graph
 86            if hasattr(self._graph, "astream")
 87            else self._graph.compile()
 88        )
 89
 90        step = 0
 91        last_event: Any = {}
 92
 93        async for node_events in compiled.astream(input):
 94            # node_events is a dict mapping node_name → node_output_state.
 95
 96            # --- Budget guard (checked *before* yielding) ---
 97            if not ctx.is_budget_ok():
 98                from harness.core.context import StepEvent
 99
100                budget_event = StepEvent(
101                    run_id=ctx.run_id,
102                    step=step,
103                    event_type="budget_exceeded",
104                    payload={
105                        "framework": self.framework_name,
106                        "step_count": ctx.step_count,
107                        "max_steps": ctx.max_steps,
108                    },
109                    timestamp=_utcnow(),
110                )
111                await self._publish(budget_event)
112                yield budget_event
113                return
114
115            # Advance the harness step counter; raises BudgetExceeded if over.
116            try:
117                ctx.tick()
118            except BudgetExceeded as exc:
119                from harness.core.context import StepEvent
120
121                budget_event = StepEvent(
122                    run_id=ctx.run_id,
123                    step=ctx.step_count,
124                    event_type="budget_exceeded",
125                    payload={
126                        "framework": self.framework_name,
127                        "error": str(exc),
128                    },
129                    timestamp=_utcnow(),
130                )
131                await self._publish(budget_event)
132                yield budget_event
133                return
134
135            step += 1
136
137            # Emit one StepEvent per active node in this chunk.
138            for node_name, node_output in node_events.items():
139                from harness.core.context import StepEvent
140
141                output_keys: list[str] = (
142                    list(node_output.keys())
143                    if isinstance(node_output, dict)
144                    else []
145                )
146
147                # Optional: log token usage if the node output exposes it.
148                token_usage = self._extract_token_usage(node_output)
149                if token_usage:
150                    logger.debug(
151                        "LangGraph node '%s' used %d tokens",
152                        node_name,
153                        token_usage,
154                    )
155
156                event = StepEvent(
157                    run_id=ctx.run_id,
158                    step=step,
159                    event_type="tool_call",
160                    payload={
161                        "framework": self.framework_name,
162                        "node": node_name,
163                        "output_keys": output_keys,
164                        "token_usage": token_usage,
165                    },
166                    timestamp=_utcnow(),
167                )
168
169                await self._publish(event)
170                yield event
171
172            last_event = node_events  # keep the most recent chunk as final state
173
174        self._final_state = last_event
175        self._step_count = step

177    async def get_result(self) -> FrameworkResult:
178        """Return the :class:`FrameworkResult` built from the graph's final state.
179
180        Raises:
181            RuntimeError: If ``run()`` has not been called yet.
182        """
183        if self._final_state is None:
184            raise RuntimeError(
185                "LangGraphAdapter.get_result() called before run() completed."
186            )
187
188        state = self._final_state
189        # The final event from astream is {node_name: output}.  Merge all node
190        # outputs to build a single state dict so we can probe common keys.
191        merged: dict[str, Any] = {}
192        if isinstance(state, dict):
193            for node_output in state.values():
194                if isinstance(node_output, dict):
195                    merged.update(node_output)
196            # Also include top-level keys (some graphs expose them directly).
197            merged.update(
198                {k: v for k, v in state.items() if not isinstance(v, dict)}
199            )
200
201        output = self._extract_output(merged) or self._extract_output(
202            state if isinstance(state, dict) else {}
203        )
204
205        return FrameworkResult(
206            framework=self.framework_name,
207            output=output,
208            steps=self._step_count,
209            metadata=merged if merged else (state if isinstance(state, dict) else {}),
210        )
